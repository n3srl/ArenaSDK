Trigger, Overlapping Trigger¶

using System;
using System.Collections.Generic; // for List<>

namespace Cs_Trigger_OverlappingTrigger
{
        class Cs_Trigger_OverlappingTrigger
        {
                const String TAB1 = "  ";
                const String TAB2 = "    ";

                // Trigger: Overlapping Trigger on Exposure End Events
                //    This example demonstrates the use of trigger and exposure end
                //    events to minimize the amount of unused exposure time between
                //    images. This is done by setting the device to start exposing (or
                //    trigger) right when the last exposure has just finished (or
                //    exposure end event). After receiving each exposure end event
                //    notification, the next trigger is executed to acquire the next
                //    image. Once all triggers have executed, the images are retrieved
                //    from the output queue. This example shows that there is little time
                //    between triggering images, and that the exposure time is close to
                //    the time between triggers.

                // =-=-=-=-=-=-=-=-=-
                // =-=- SETTINGS =-=-
                // =-=-=-=-=-=-=-=-=-

                // Number of images to grab
                const Int32 NUM_IMAGES = 10;

                // Event timeout
                const Int32 EVENT_TIMEOUT = 3000;

                // image timeout
                const Int32 IMAGE_TIMEOUT = 2000;

                // system timeout
                const Int32 SYSTEM_TIMEOUT = 100;

                // =-=-=-=-=-=-=-=-=-
                // =-=- EXAMPLE -=-=-
                // =-=-=-=-=-=-=-=-=-

                // demonstrates basic trigger configuration and use
                // (1) sets trigger mode, source, and selector
                // (2) sets acquisition mode, event selector, and turns on event notifications
                // (3) starts stream
                // (4) waits on trigger armed, executes trigger, and waits on exposureEnd event for all images
                // (5) grabs each image and requeues image buffer for all images
                // (6) stops stream
                static void OverlapTriggerOnExposureEndEvent(ArenaNET.IDevice device)
                {
                        // Initialize events
                        //    Events nodes aren't available until the events engine has been
                        //    initialized.
                        device.InitializeEvents();

                        // get node values that will be changed in order to return their
                        // values at the end of the example
                        var triggerSelectorNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("TriggerSelector");
                        var triggerModeNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("TriggerMode");
                        var triggerSourceNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("TriggerSource");
                        var triggerOverlapNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("TriggerOverlap");
                        var acquisitionModeNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("AcquisitionMode");
                        var eventSelectorNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("EventSelector");
                        var eventNotificationNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("EventNotification");
                        var exposureAutoNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("ExposureAuto");

                        var triggerSelectorInitial = triggerSelectorNode.IntValue;
                        var triggerModeInitial = triggerModeNode.IntValue;
                        var triggerSourceInitial = triggerSourceNode.IntValue;
                        var triggerOverlapInitial = triggerOverlapNode.IntValue;
                        var acquisitionModeInitial = acquisitionModeNode.IntValue;
                        var eventSelectorInitial = eventSelectorNode.IntValue;
                        var eventNotificationInitial = eventNotificationNode.IntValue;
                        var exposureAutoInitial = exposureAutoNode.IntValue;

                        // Set trigger selector
                        //    Set the trigger selector to FrameStart. When triggered, the
                        //    device will start acquiring a single frame. This can also be
                        //    set to AcquisitionStart or FrameBurstStart.
                        Console.WriteLine("{0}Set trigger selector to FrameStart", TAB1);

                        triggerSelectorNode.FromString("FrameStart");

                        // Set trigger mode
                        //    Enable trigger mode before setting the source and selector and
                        //    before starting the stream. Trigger mode cannot be turned on
                        //    and off while the device is streaming.
                        Console.WriteLine("{0}Enable trigger mode", TAB1);

                        triggerModeNode.FromString("On");

                        // Set trigger source
                        //    Set the trigger source to software in order to trigger images
                        //    without the use of any additional hardware. Lines of the GPIO
                        //    can also be used to trigger.
                        Console.WriteLine("{0}Set trigger source to Software", TAB1);

                        triggerSourceNode.FromString("Software");

                        var triggerSoftwareNode = (ArenaNET.ICommand)device.NodeMap.GetNode("TriggerSoftware");

                        // Set trigger overlap
                        //    Trigger overlap defines when a trigger can start accepting a
                        //    new frame. Setting trigger to overlap with the previous frame
                        //    allows the camera to being exposing the new frame while the
                        //    camera is still reading out the sensor data acquired from the
                        //    previous frame.
                        Console.WriteLine("{0}Set trigger overlap to PreviousFrame", TAB1);

                        triggerOverlapNode.FromString("PreviousFrame");

                        // Set acquisition mode
                        //    When trigger mode is off and the acquisition mode is set to
                        //    stream continuously, starting the stream will have the camera
                        //    begin acquiring a steady stream of images. However, with
                        //    trigger mode enabled, the device will wait for the trigger
                        //    before acquiring any.
                        Console.WriteLine("{0}Set acquisiton mode to: \"Continuous\"", TAB1);

                        acquisitionModeNode.FromString("Continuous");

                        // Set event selector
                        //    We want to trigger and wait to be notified as soon as a certain
                        //    event occurs while making the image. Here we choose to be
                        //    notified at the end of the exposure of an image.
                        Console.WriteLine("{0}Set event selector to: \"ExposureEnd\"", TAB1);

                        eventSelectorNode.FromString("ExposureEnd");

                        // Set event notification
                        //    After choosing which event to be notified about, enabling the
                        //    EventNotification node will turn on the notification for the
                        //    event selected in the EventSelector node.
                        Console.WriteLine("{0}Enable event notifications for \"ExposureEnd\"", TAB1);

                        eventNotificationNode.FromString("On");

                        // Retrieve exposure time
                        //    The exposure time is similar to the time between triggering
                        //    images. This is shown by turning off automatic exposure and
                        //    retrieving the exposure time.
                        exposureAutoNode.FromString("Off");

                        var exposureTimeNode = device.NodeMap.GetNode("ExposureTime") as ArenaNET.IFloat;
                        double exposureTime = exposureTimeNode.Value;

                        // convert to nanoseconds
                        double exposureTimeNs = exposureTime * 1000;
                        Console.WriteLine("{0}Exposure time: {1} ns", TAB1, exposureTimeNs);

                        // number of images to grab
                        const UInt32 numberOfImages = NUM_IMAGES;

                        // timeout constants
                        const Int64 waitOnEventTimeout = EVENT_TIMEOUT;
                        const Int64 imageTimeout = IMAGE_TIMEOUT;

                        // enable stream auto negotiate packet size
                        var streamAutoNegotiatePacketSizeNode = (ArenaNET.IBoolean)device.TLStreamNodeMap.GetNode("StreamAutoNegotiatePacketSize");
                        streamAutoNegotiatePacketSizeNode.Value = true;

                        // enable stream packet resend
                        var streamPacketResendEnableNode = (ArenaNET.IBoolean)device.TLStreamNodeMap.GetNode("StreamPacketResendEnable");
                        streamPacketResendEnableNode.Value = true;

                        // start stream
                        Console.WriteLine("{0}Start stream with {1} buffers", TAB1, numberOfImages);
                        device.StartStream(numberOfImages);

                        // Trigger an image
                        //    Trigger an image manually, since trigger mode is enabled. This
                        //    triggers the camera to acquire a single image. A buffer is then
                        //    filled and moved to the output queue, where it will wait to be
                        //    retrieved.
                        for (Int32 i = 0; i < numberOfImages; i++)
                        {
                                Console.Write("{0}Image {1} triggered", TAB2, i.ToString().PadLeft(4));

                                // Trigger Armed
                                //    Continually check until trigger is armed. Once the trigger
                                //    is armed, it is ready to be executed.
                                Boolean isTriggerArmed = false;
                                do
                                {
                                        var triggerArmedNode = (ArenaNET.IBoolean)device.NodeMap.GetNode("TriggerArmed");

                                        isTriggerArmed = triggerArmedNode.Value;

                                } while (isTriggerArmed == false);

                                triggerSoftwareNode.Execute();

                                // Wait on event
                                //    Wait on event to process before continuing. The data is
                                //    created from the event generation, not from waiting on it.
                                device.WaitOnEvent(waitOnEventTimeout);

                                Console.Write(" and ExposureEnd Event notification arrived\r");
                        }

                        Console.WriteLine("\n{0}Done triggering {1} images", TAB1, numberOfImages.ToString().PadLeft(4));

                        // retrieve images and timestamps
                        List<double> timestampNs = new List<double>();

                        for (Int32 i = 0; i < numberOfImages; i++)
                        {
                                ArenaNET.IImage image = device.GetImage(imageTimeout);
                                timestampNs.Add(image.TimestampNs);

                                Console.Write("{0}Image timestamp: {1} ns", TAB2, image.TimestampNs.ToString().PadLeft(4));

                                if (i > 0)
                                {
                                        // calculate time between triggers
                                        double triggerTimestampDifference = timestampNs[i] - timestampNs[i - 1];

                                        Console.WriteLine(" ({0} ns since last trigger)", triggerTimestampDifference);

                                }
                                else
                                        Console.WriteLine("");

                                // requeue buffer for each image
                                device.RequeueBuffer(image);
                        }

                        // stop stream
                        Console.WriteLine("{0}Stop stream\n", TAB1);

                        device.StopStream();

                        // Deinitialize events
                        //    Deinitialize the events engine in order to deallocate memory
                        //    and stop processing events. Failing to deinitialize events
                        //    results in a memory leak.
                        device.DeinitializeEvents();

                        // return nodes to their initial values
                        exposureAutoNode.IntValue = exposureAutoInitial;
                        eventNotificationNode.IntValue = eventNotificationInitial;
                        eventSelectorNode.IntValue = eventSelectorInitial;
                        acquisitionModeNode.IntValue = acquisitionModeInitial;
                        triggerOverlapNode.IntValue = triggerOverlapInitial;
                        triggerSourceNode.IntValue = triggerSourceInitial;
                        triggerModeNode.IntValue = triggerModeInitial;
                        triggerSelectorNode.IntValue = triggerSelectorInitial;
                }

                // =-=-=-=-=-=-=-=-=-
                // =- PREPARATION -=-
                // =- & CLEAN UP =-=-
                // =-=-=-=-=-=-=-=-=-

                static void Main(string[] args)
                {
                        Console.WriteLine("Cs_Trigger_OverlappingTrigger");

                        try
                        {
                                // prepare example
                                ArenaNET.ISystem system = ArenaNET.Arena.OpenSystem();
                                system.UpdateDevices(100);
                                if (system.Devices.Count == 0)
                                {
                                        Console.WriteLine("\nNo camera connected\nPress enter to complete");
                                        Console.Read();
                                        return;
                                }
                                ArenaNET.IDevice device = system.CreateDevice(system.Devices[0]);

                                // run example
                                Console.WriteLine("Commence example\n");
                                OverlapTriggerOnExposureEndEvent(device);
                                Console.WriteLine("\nExample complete");

                                // clean up example
                                system.DestroyDevice(device);
                                ArenaNET.Arena.CloseSystem(system);
                        }
                        catch (Exception ex)
                        {
                                Console.WriteLine("Exception thrown: {0}", ex.Message);
                        }

                        Console.WriteLine("Press enter to complete");
                        Console.Read();
                }
        }
}
