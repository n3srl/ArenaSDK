﻿using System;
using System.Collections.Generic; // for List<>

namespace Cs_Streamables
{
        class Cs_Streamables
        {
                const String TAB1 = "  ";

                // Streamables
                //    This example introduces streamables, which uses files to pass
                //    settings around between devices. This example writes all streamable
                //    features from a source device to a file, and then writes them from
                //    the file to all other connected devices.

                // =-=-=-=-=-=-=-=-=-
                // =-=- SETTINGS =-=-
                // =-=-=-=-=-=-=-=-=-

                // the name of the file to stream features to/from
                const String FILE_NAME = "allStreamableFeatures.txt";

                // =-=-=-=-=-=-=-=-=-
                // =-=- EXAMPLE -=-=-
                // =-=-=-=-=-=-=-=-=-

                // demonstrates streamable features
                // (1) reads all streamable features from source device
                // (2) writes features to file
                // (3) reads features from file
                // (4) writes features to destination devices
                static void WriteAndReadStreamables(ArenaNET.IDevice srcDevice, List<ArenaNET.IDevice> dstDevices)
                {
                        // get device serial
                        String srcDeviceSerialNumber = (srcDevice.NodeMap.GetNode("DeviceSerialNumber") as ArenaNET.IString).Value;

                        // Write features to file
                        //    Write streamable features from a device to a file. Each node
                        //    map requires a separate feature stream object. When writing to
                        //    a file, if no features are explicitly selected, all will be
                        //    written by default.
                        Console.WriteLine("{0}Save features from device {1} to {2}", TAB1, srcDeviceSerialNumber, FILE_NAME);

                        ArenaNET.FeatureStream featureStreamSrc = new ArenaNET.FeatureStream(srcDevice.NodeMap);
                        featureStreamSrc.Write(FILE_NAME);

                        // Read features to devices
                        //    Read streamable features from from a file to the rest of the
                        //    devices. Again, each node map requires a separate feature
                        //    stream object. When reading from a file, all features saved to
                        //    the file will be loaded to the device. If a device does not
                        //    have a feature, it is skipped.
                        foreach (ArenaNET.IDevice device in dstDevices)
                        {
                                String deviceSerialNumber = (device.NodeMap.GetNode("DeviceSerialNumber") as ArenaNET.IString).Value;

                                Console.WriteLine("{0}Load features from {1} to device {2}", TAB1, FILE_NAME, deviceSerialNumber);

                                ArenaNET.FeatureStream featureStreamDst = new ArenaNET.FeatureStream(device.NodeMap);
                                featureStreamDst.Read(FILE_NAME);
                        }
                }

                static void Main(string[] args)
                {
                        Console.WriteLine("Cs_Streamables");
                        Console.Write("Example may change device settings -- proceed? ('y' to continue) ");
                        string continueExample = Console.ReadLine();

                        if (continueExample.Equals("y"))
                        {
                                try
                                {
                                        // prepare example
                                        ArenaNET.ISystem system = ArenaNET.Arena.OpenSystem();
                                        system.UpdateDevices(100);
                                        if (system.Devices.Count == 0)
                                        {
                                                Console.WriteLine("\nNo camera connected\nPress enter to complete");
                                                Console.Read();
                                                return;
                                        }
                                        else if (system.Devices.Count == 1)
                                        {
                                                Console.WriteLine("\nWarning: only one device connected. Example runs best with at least 2 devices");
                                        }
                                        ArenaNET.IDevice srcDevice = null;
                                        List<ArenaNET.IDevice> dstDevices = new List<ArenaNET.IDevice>();
                                        Int16 i = 0;
                                        foreach (ArenaNET.IDeviceInfo deviceInfo in system.Devices)
                                        {
                                                ArenaNET.IDevice device = system.CreateDevice(deviceInfo);
                                                if (i++ == 0)
                                                {
                                                        srcDevice = device;
                                                }
                                                else
                                                {
                                                        dstDevices.Add(device);
                                                }
                                        }

                                        // run example
                                        Console.WriteLine("Commence example\n");
                                        WriteAndReadStreamables(srcDevice, dstDevices);
                                        Console.WriteLine("\nExample complete");

                                        // clean up example
                                        system.DestroyDevice(srcDevice);
                                        foreach (ArenaNET.IDevice device in dstDevices)
                                        {
                                                system.DestroyDevice(device);
                                        }
                                        ArenaNET.Arena.CloseSystem(system);
                                }
                                catch (Exception ex)
                                {
                                        Console.WriteLine("An error occurred: {0}", ex.Message);
                                }
                        }

                        Console.WriteLine("Press enter to complete ");
                        Console.Read();
                }
        }
}
