using System;

namespace Cs_Callback_OnNodeChange
{
        class Cs_Callback_OnNodeChange
        {
                const String TAB1 = "  ";
                const String TAB2 = "    ";
                const String TAB3 = "      ";
                const String TAB4 = "        ";

                // Callbacks: On Node Change
                //    This example demonstrates configuring a callback to be invoked when
                //    a node is invalidated. A node is invalidated when its value changes
                //    or can be invalidated manually. In this example, a callback is
                //    registered on PayloadSize. The example shows two ways to invoke a
                //    callback: first by changing the value of a dependent node (Height)
                //    and then by invalidating PayloadSize manually. Whenever the
                //    callback is triggered, the callback function prints the updated
                //    value of the invalidated node

                // =-=-=-=-=-=-=-=-=-
                // =-=- SETTINGS =-=-
                // =-=-=-=-=-=-=-=-=-

                // Height values
                //    The example changes the value of the height node twice to ensure
                //    that the height is changed.
                const UInt32 HEIGHT_ONE = 256;
                const UInt32 HEIGHT_TWO = 512;

                // =-=-=-=-=-=-=-=-=-
                // =-=- EXAMPLE -=-=-
                // =-=-=-=-=-=-=-=-=-

                // callback function
                // (1) prints information from the callback
                static void PrintIntegerNodeValue(ArenaNET.INode node)
                {
                        // Do something on callback
                        //    This function is registered with the callback and invoked when
                        //    the registered node is invalidated. In this case, the callback
                        //    prints the display name and value of the invalidated node.
                        ArenaNET.IValue value = (ArenaNET.IValue)node;

                        Console.WriteLine("{0}Message from callback", TAB3);
                        Console.WriteLine("{0}{1}: {2}", TAB4, node.DisplayName, value.ToString());
                }

                // demonstrates callbacks invoked on node changes
                // (1) registers callback on node PayloadSize
                // (2) changes Height twice to invalidate PayloadSize, invoking callback
                // (3) invalidates PayloadSize manually
                // (4) deregisters callback
                static void ConfigureAndCauseCallback(ArenaNET.IDevice device)
                {
                        // get node values that will be changed in order to return their
                        // values at the end of the example
                        var heightNode = (ArenaNET.IInteger)device.NodeMap.GetNode("Height");
                        Int64 heightInitial = heightNode.Value;

                        // Register PayloadSize for callbacks
                        //    Callbacks are registered with a node and a function. This
                        //    example demonstrates callbacks being invoked when the node is
                        //    invalidated. This could be when the node value changes, either
                        //    manually or by the device, or the node is invalidated manually.
                        Console.WriteLine("{0}Register callback on PayloadSize", TAB1);

                        // get node
                        var payloadSizeNode = device.NodeMap.GetNode("PayloadSize") as ArenaNET.IInteger;

                        if (payloadSizeNode.Equals(null) || !payloadSizeNode.IsReadable)
                                throw new ArenaNET.GenericException("PayloadSize node not found/readable");

                        // create delegate for registering callback
                        ArenaNET.INode.del d = new ArenaNET.INode.del(PrintIntegerNodeValue);

                        // register callback
                        IntPtr callback = payloadSizeNode.RegisterCallback(d);

                        // Modify Height to invoke callback on PayloadSize
                        //    The value of PayloadSize depends on a number of other nodes.
                        //    This includes Height. Therefore, changing the value of Height
                        //    changes the value of and invalidates PayloadSize, which then
                        //    invokes the callback.
                        Console.WriteLine("{0}Change height once", TAB2);

                        heightNode.Value = HEIGHT_ONE;

                        Console.WriteLine("{0}Change height twice", TAB2);

                        heightNode.Value = HEIGHT_TWO;

                        // Manually invalidate PayloadSize for callback
                        //    Apart from changing the value of a node, nodes can be
                        //    invalidated manually by calling InvalidateNode. This also
                        //    invokes the callback.
                        Console.WriteLine("{0}Invalidate PayloadSize", TAB2);

                        payloadSizeNode.InvalidateNode();

                        // Deregister callback
                        //    Failing to deregister a callback results in a memory leak. Once
                        //    a callback has been registered, it will no longer be invoked
                        //    when a node is invalidated.
                        Console.WriteLine("{0}Deregister callback", TAB1);

                        payloadSizeNode.DeregisterCallback(callback);

                        // return nodes to their initial values
                        heightNode.Value = heightInitial;
                }

                // =-=-=-=-=-=-=-=-=-
                // =- PREPARATION -=-
                // =- & CLEAN UP =-=-
                // =-=-=-=-=-=-=-=-=-

                static void Main(string[] args)
                {
                        Console.WriteLine("Cs_Callback_OnNodeChange");

                        try
                        {
                                // prepare example
                                ArenaNET.ISystem system = ArenaNET.Arena.OpenSystem();
                                system.UpdateDevices(100);
                                if (system.Devices.Count == 0)
                                {
                                        Console.WriteLine("\nNo camera connected\nPress enter to complete");
                                        Console.Read();
                                        return;
                                }
                                ArenaNET.IDevice device = system.CreateDevice(system.Devices[0]);

                                // run example
                                Console.WriteLine("Commence example\n");
                                ConfigureAndCauseCallback(device);
                                Console.WriteLine("\nExample complete");

                                // clean up example
                                system.DestroyDevice(device);
                                ArenaNET.Arena.CloseSystem(system);
                        }
                        catch (Exception ex)
                        {
                                Console.WriteLine("Exception thrown: {0}", ex.Message);
                        }

                        Console.WriteLine("Press enter to complete");
                        Console.Read();
                }
        }
}
