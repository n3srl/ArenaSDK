using System;
using System.Collections.Generic; // for List<>

namespace Cs_Explore_Nodes
{
        class Cs_Explore_Nodes
        {
                // Explore: Nodes
                //    This example explores traversing the nodes as a tree and fundamental node
                //    information including display name, node name, access mode visibility,
                //        interface type, and value.

                // =-=-=-=-=-=-=-=-=-
                // =-=- SETTINGS =-=-
                // =-=-=-=-=-=-=-=-=-

                // Choose node properties to explore
                const Boolean EXPLORE_ACCESS = true;
                const Boolean EXPLORE_VISIBILITY = true;
                const Boolean EXPLORE_TYPE = true;
                const Boolean EXPLORE_VALUE = true;

                // =-=-=-=-=-=-=-=-=-
                // =-=- EXAMPLE -=-=-
                // =-=-=-=-=-=-=-=-=-

                // maintains display indentation
                static String Indent(UInt32 depth)
                {
                        String indentation = "  ";
                        for (int i = 0; i < depth; i++)
                                indentation += "  ";

                        return indentation;
                }

                // explores node
                // (1) retrieves display name
                // (2) retrieves node name
                // (3) retrieves accessibility
                // (4) retrieves visibility
                // (5) retrieves interface type
                // (6) retrieves value
                static void ExploreNode(ArenaNET.INode node, uint depth = 0)
                {
                        // Retrieve display name
                        String displayName = node.DisplayName;

                        // Retrieve node name
                        String nodeName = node.Name;

                        // Retrieve accessibility
                        String accessModeStr = node.AccessMode.ToString();

                        // Retrieve visibility
                        String visibilityStr = node.Visibility.ToString();

                        // Retrieve interface type
                        String interfaceTypeStr = node.PrincipalInterfaceType.ToString();

                        // Retrieve value
                        String value = "-";
                        if (node.IsReadable)
                                value = node.ToString();

                        // print node information
                        String displayAndNodeName = displayName + " (" + nodeName + ")";
                        Console.Write("{0}{1}", Indent(depth), displayAndNodeName.PadRight(90 - ((int)depth * 2)));

                        if (EXPLORE_ACCESS)
                                Console.Write("{0}", accessModeStr.PadRight(7));

                        if (EXPLORE_VISIBILITY)
                                Console.Write("{0}", visibilityStr.PadRight(14));

                        if (EXPLORE_TYPE)
                                Console.Write("{0}", interfaceTypeStr.PadRight(20));

                        if (EXPLORE_VALUE)
                                Console.Write("{0}", value.Length < 90 ? value : "...");

                        Console.Write("\n");

                        // Explore category node children
                        if (node.PrincipalInterfaceType == ArenaNET.EInterfaceType.intfICategory)
                        {
                                // cast node as ICategory here in order to access Features
                                ArenaNET.ICategory category = node as ArenaNET.ICategory;
                                List<ArenaNET.IValue> children = category.Features;
                                foreach (ArenaNET.IValue val in children)
                                        ExploreNode(val.Node, depth + 1);
                        }
                }

                // =-=-=-=-=-=-=-=-=-
                // =- PREPARATION -=-
                // =- & CLEAN UP =-=-
                // =-=-=-=-=-=-=-=-=-

                static void Main(string[] args)
                {
                        Console.WriteLine("Cs_Explore_Nodes");

                        try
                        {
                                // prepare example
                                ArenaNET.ISystem system = ArenaNET.Arena.OpenSystem();
                                system.UpdateDevices(100);
                                if (system.Devices.Count == 0)
                                {
                                        Console.WriteLine("\nNo camera connected\nPress enter to complete");
                                        Console.Read();
                                        return;
                                }

                                ArenaNET.IDevice pDevice = system.CreateDevice(system.Devices[0]);

                                ArenaNET.INode pDeviceRoot = pDevice.NodeMap.GetNode("Root");
                                ArenaNET.INode pTLDeviceRoot = pDevice.TLDeviceNodeMap.GetNode("Root");
                                ArenaNET.INode pTLStreamRoot = pDevice.TLStreamNodeMap.GetNode("Root");
                                ArenaNET.INode pTLInterfaceRoot = pDevice.TLInterfaceNodeMap.GetNode("Root");
                                ArenaNET.INode pTLSystemRoot = system.TLSystemNodeMap.GetNode("Root");

                                // run example
                                Console.WriteLine("Commence example");

                                Console.WriteLine("\nDevice Nodemap");
                                ExploreNode(pDeviceRoot);

                                Console.WriteLine("\nTL Device Nodemap");
                                ExploreNode(pTLDeviceRoot);

                                Console.WriteLine("\nTL Stream Nodemap");
                                ExploreNode(pTLStreamRoot);

                                Console.WriteLine("\nTL Interface Nodemap");
                                ExploreNode(pTLInterfaceRoot);

                                Console.WriteLine("\nTL System Nodemap");
                                ExploreNode(pTLSystemRoot);

                                Console.WriteLine("\nExample complete");

                                // clean up example
                                system.DestroyDevice(pDevice);
                                ArenaNET.Arena.CloseSystem(system);
                        }
                        catch (Exception ex)
                        {
                                Console.WriteLine("Exception thrown: {0}", ex.Message);
                        }

                        Console.WriteLine("Press enter to complete");
                        Console.Read();
                }
        }
}
