Exposure, For HDR¶

using System;
using System.Collections.Generic; // for List<>

namespace Cs_Exposure_ForHDR
{
        class Cs_Exposure_ForHDR
        {
                const String TAB1 = "  ";
                const String TAB2 = "    ";
                const String TAB3 = "      ";

                // Exposure: For High Dynamic Range
                //    This example demonstrates dynamically updating the exposure time in
                //    order to grab images appropriate for high dynamic range (or HDR)
                //    imaging. HDR images can be created by combining a number of images
                //    acquired at various exposure times. This example demonstrates
                //    grabbing three images for this purpose, without the actual creation
                //    of an HDR image.

                // =-=-=-=-=-=-=-=-=-
                // =-=- SETTINGS =-=-
                // =-=-=-=-=-=-=-=-=-

                // Exposure times
                static double EXPOSURE_HIGH = 100000.0;
                static double EXPOSURE_MID = 50000.0;
                static double EXPOSURE_LOW = 25000.0;

                // Image timeout
                //    Timeout for grabbing images (in milliseconds). Have the timeout at
                //    least a bit larger than the highest exposure time to avoid timing
                //    out.
                const UInt64 TIMEOUT = ArenaNET.Consts.INF;

                // number of images to grab
                const UInt32 NUM_HDR_IMAGES = 5;

                // =-=-=-=-=-=-=-=-=-
                // =-=- EXAMPLE -=-=-
                // =-=-=-=-=-=-=-=-=-

                // HDR image struct
                //    This struct holds the information needed for a single HDR image.
                //    This includes one image taken with a high exposure time, one with a
                //    medium exposure time, and one with a fairly low exposure time.
                struct HDRImage
                {
                        public ArenaNET.IImage imageHigh, imageMid, imageLow;
                }

                // Trigger software once armed
                //    Continually check until trigger is armed. Once the trigger is
                //    armed, it is ready to be executed.
                static void TriggerSoftwareOnceArmed(ArenaNET.IDevice device)
                {
                        // wait until trigger armed is true
                        bool triggerArmed = false;
                        do
                        {
                                triggerArmed = ((ArenaNET.IBoolean)device.NodeMap.GetNode("TriggerArmed")).Value;
                        } while (triggerArmed == false);

                        // retrieve and execute software trigger node
                        var triggerSoftwareNode = device.NodeMap.GetNode("TriggerSoftware") as ArenaNET.ICommand;
                        triggerSoftwareNode.Execute();
                }

                // demonstrates exposure configuration and acquisition for HDR imaging
                // (1) sets trigger mode
                // (2) disables automatic exposure
                // (3) sets high exposure time
                // (4) gets first image
                // (5) sets medium exposure time
                // (6) gets second image
                // (7) sets low exposure time
                // (8) gets third images
                // (9) copies images into object for later processing
                // (10) does NOT process copied images
                // (11) cleans up copied images
                static void AcquireHDRImages(ArenaNET.IDevice device)
                {
                        // get node values that will be changed in order to return their
                        // values at the end of the example
                        var triggerModeNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("TriggerMode");
                        String triggerModeInitial = triggerModeNode.ToString();

                        var triggerSourceNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("TriggerSource");
                        String triggerSourceInitial = triggerSourceNode.ToString();

                        var triggerSelectorNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("TriggerSelector");
                        String triggerSelectorInitial = triggerSelectorNode.ToString();

                        var exposureAutoNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("ExposureAuto");
                        String exposureAutoInitial = exposureAutoNode.ToString();

                        // Get exposure time and software trigger nodes
                        //    The exposure time and software trigger nodes are retrieved
                        //    beforehand in order to check for existance, readability, and
                        //    writability only once before the stream.
                        Console.WriteLine("{0}Get exposure time and software trigger nodes", TAB1);

                        var exposureTimeNode = device.NodeMap.GetNode("ExposureTime") as ArenaNET.IFloat;
                        double exposureTimeInitial = exposureTimeNode.Value;

                        var triggerSoftwareNode = device.NodeMap.GetNode("TriggerSoftware") as ArenaNET.ICommand;

                        // Prepare trigger mode
                        //    Enable trigger mode before starting the stream. This example
                        //    uses the trigger to control the moment that images are taken.
                        //    This ensures the exposure time of each image in a way that a
                        //    continuous stream might have trouble with.
                        Console.WriteLine("{0}Prepare trigger mode", TAB1);

                        triggerSelectorNode.FromString("FrameStart");
                        triggerModeNode.FromString("On");
                        triggerSourceNode.FromString("Software");

                        // Disable automatic exposure
                        //    Disable automatic exposure before starting the stream. The HDR
                        //    images in this example require three images of varied
                        //    exposures, which need to be set manually.
                        Console.WriteLine("{0}Disable automatic exposure", TAB1);

                        exposureAutoNode.FromString("Off");

                        if (exposureTimeNode.Equals(null) || triggerSoftwareNode.Equals(null))
                                throw new ArenaNET.GenericException("ExposureTime or TriggerSoftware node not found");

                        if (!exposureTimeNode.IsWritable || !triggerSoftwareNode.IsWritable)
                                throw new ArenaNET.GenericException("ExposureTime or TriggerSoftware node not writable");

                        // enable stream auto negotiate packet size
                        var streamAutoNegotiatePacketSizeNode = (ArenaNET.IBoolean)device.TLStreamNodeMap.GetNode("StreamAutoNegotiatePacketSize");
                        streamAutoNegotiatePacketSizeNode.Value = true;

                        // enable stream packet resend
                        var streamPacketResendEnableNode = (ArenaNET.IBoolean)device.TLStreamNodeMap.GetNode("StreamPacketResendEnable");
                        streamPacketResendEnableNode.Value = true;

                        // start stream
                        device.StartStream();

                        // get images for HDR
                        Console.WriteLine("{0}Acquire HDR images", TAB1);

                        List<HDRImage> hdrImages = new List<HDRImage>();

                        for (int i = 0; i < NUM_HDR_IMAGES; i++)
                        {
                                // Get high, medium, and low exposure images
                                //    This example grabs three examples of varying exposures for
                                //    later processing. For each image, the exposure must be set,
                                //    an image must be triggered, and then that image must be
                                //    retrieved. After the exposure time is changed, the setting
                                //    does not take place on the device until after the next
                                //    frame. Because of this, two images are retrieved, the first
                                //    of which is discarded.
                                Console.WriteLine("{0}Get HDR image {1}", TAB2, i);

                                // get max and min exposure time to ensure set of exposure times
                                // are within this range
                                double exposureTimeMax = exposureTimeNode.Max;
                                double exposureTimeMin = exposureTimeNode.Min;

                                // if largest exposure times is not within the exposure time
                                // range, set largest exposure time to max value and set the
                                // remaining exposure times to half the value of the state before
                                if (EXPOSURE_HIGH > exposureTimeMax || EXPOSURE_LOW < exposureTimeMin)
                                {
                                        EXPOSURE_HIGH = exposureTimeMax;
                                        EXPOSURE_MID = EXPOSURE_HIGH / 2;
                                        EXPOSURE_LOW = EXPOSURE_MID / 2;
                                }

                                // high exposure image
                                exposureTimeNode.Value = EXPOSURE_HIGH;
                                TriggerSoftwareOnceArmed(device);
                                ArenaNET.IImage imagePreHigh = device.GetImage(TIMEOUT);
                                TriggerSoftwareOnceArmed(device);
                                ArenaNET.IImage imageHigh = device.GetImage(TIMEOUT);

                                Console.WriteLine("{0}High image (timestamp {1}, exposure {2})", TAB3, imageHigh.TimestampNs, EXPOSURE_HIGH);

                                // medium exposure image
                                exposureTimeNode.Value = EXPOSURE_MID;
                                TriggerSoftwareOnceArmed(device);
                                ArenaNET.IImage imagePreMid = device.GetImage(TIMEOUT);
                                TriggerSoftwareOnceArmed(device);
                                ArenaNET.IImage imageMid = device.GetImage(TIMEOUT);

                                Console.WriteLine("{0}Mid image (timestamp {1}, exposure {2})", TAB3, imageMid.TimestampNs, EXPOSURE_MID);

                                // low exposure image
                                exposureTimeNode.Value = EXPOSURE_LOW;
                                TriggerSoftwareOnceArmed(device);
                                ArenaNET.IImage imagePreLow = device.GetImage(TIMEOUT);
                                TriggerSoftwareOnceArmed(device);
                                ArenaNET.IImage imageLow = device.GetImage(TIMEOUT);

                                Console.WriteLine("{0}Low image (timestamp {1}, exposure {2})", TAB3, imageLow.TimestampNs, EXPOSURE_LOW);

                                // Copy images for processing later
                                //    Use the image factory to copy the images for later
                                //    processing. Images are copied in order to requeue buffers
                                //    to allow for more images to be retrieved from the device.
                                Console.WriteLine("{0}Copy images for HDR processing later", TAB2, i);

                                HDRImage hdrImage = new HDRImage();
                                hdrImage.imageHigh = ArenaNET.ImageFactory.Copy(imageHigh);
                                hdrImage.imageMid = ArenaNET.ImageFactory.Copy(imageMid);
                                hdrImage.imageLow = ArenaNET.ImageFactory.Copy(imageLow);
                                hdrImages.Add(hdrImage);

                                device.RequeueBuffer(imagePreHigh);
                                device.RequeueBuffer(imageHigh);
                                device.RequeueBuffer(imagePreMid);
                                device.RequeueBuffer(imageMid);
                                device.RequeueBuffer(imagePreLow);
                                device.RequeueBuffer(imageLow);
                        }

                        // stop stream
                        device.StopStream();

                        // Run HDR processing
                        //    Once the images have been retrieved and copied, they can be
                        //    processed into an HDR image. HDR algorithms
                        Console.WriteLine("{0}Run HDR processing", TAB1);

                        // ...
                        // ...
                        // ...

                        // clean up copied images
                        for (int i = 0; i < hdrImages.Count; i++)
                        {
                                ArenaNET.ImageFactory.Destroy(hdrImages[i].imageHigh);
                                ArenaNET.ImageFactory.Destroy(hdrImages[i].imageMid);
                                ArenaNET.ImageFactory.Destroy(hdrImages[i].imageLow);
                        }

                        // return nodes to their initial values
                        exposureTimeNode.Value = exposureTimeInitial;
                        exposureAutoNode.FromString(exposureAutoInitial);
                        triggerSelectorNode.FromString(triggerSelectorInitial);
                        triggerSourceNode.FromString(triggerSourceInitial);
                        triggerModeNode.FromString(triggerModeInitial);
                }

                // =-=-=-=-=-=-=-=-=-
                // =- PREPARATION -=-
                // =- & CLEAN UP =-=-
                // =-=-=-=-=-=-=-=-=-

                static void Main(string[] args)
                {
                        Console.WriteLine("Cs_Exposure_ForHDR");

                        try
                        {
                                // prepare example
                                ArenaNET.ISystem system = ArenaNET.Arena.OpenSystem();
                                system.UpdateDevices(100);
                                if (system.Devices.Count == 0)
                                {
                                        Console.WriteLine("\nNo camera connected\nPress enter to complete");
                                        Console.Read();
                                        return;
                                }
                                ArenaNET.IDevice device = system.CreateDevice(system.Devices[0]);

                                // run example
                                Console.WriteLine("Commence example\n");
                                AcquireHDRImages(device);
                                Console.WriteLine("\nExample complete");

                                // clean up example
                                system.DestroyDevice(device);
                                ArenaNET.Arena.CloseSystem(system);
                        }
                        catch (Exception ex)
                        {
                                Console.WriteLine("Exception thrown: {0}", ex.Message);
                        }

                        Console.WriteLine("Press enter to complete");
                        Console.Read();
                }
        }
}
