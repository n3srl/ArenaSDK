﻿using System;
using System.Collections.Generic; // for List<>

namespace Cs_Record
{
        class Cs_Record
        {
                const String TAB1 = "  ";
                const String TAB2 = "    ";
                const String ERASE_LINE = "\t\t\t\t\t";

                // Record: Introduction
                //    This example introduces the basics of video recording using the
                //    save library. This includes preparing, creating, and writing images
                //    to the video.

                // =-=-=-=-=-=-=-=-=-
                // =-=- SETTINGS =-=-
                // =-=-=-=-=-=-=-=-=-

                // Memory Limitations
                //    This example appends many images to record a video stream and can
                //    quickly exceed memory limits when running on a "Win32" platform
                //    target. The 32-bit Windows system allocates a total of 2GB of
                //    memory for all data types. It is suggested to run this example on
                //    an "x64" platform, which has an 8TB memory limit: 2GB of shared
                //    data between the stack and static data, and the remainder for
                //    dynamic data. If running on a Win32 platform, this example reduces
                //    the height and width for each image being saved to the video file.
                const Int32 WIDTH = 800;  // image width
                const Int32 HEIGHT = 600; // image height

                // Frames per second
                //    This example chooses its FPS setting arbitrarily. If the device is
                //    outputting images at a similar frame rate, it may seem as though
                //    the image is configured for real-time video, but this is only
                //    incidental. 250 images and 25 frames per second will produce a 10
                //    second video.
                const double FRAMES_PER_SECOND = 25.0;

                // number of images to grab
                const UInt32 NUM_IMAGES = 250;

                // File name
                //    The relative path and file name to save to. After running the
                //    example, a video should exist at the location specified. The image
                //    writer chooses the file format by the image's extension. Aside from
                //    MPEG-4 (.mp4), videos can be saved as AVI (.avi), MOV (.mov), and
                //    raw (.raw) files.
                const String FILE_NAME = "Cs_Record/video.mp4";

                // =-=-=-=-=-=-=-=-=-
                // =-=- EXAMPLE -=-=-
                // =-=-=-=-=-=-=-=-=-

                static void Usage()
                {
                        Console.WriteLine("Usage:");
                        Console.WriteLine("{0} [-w width] [-h height] [-n numImages]", System.AppDomain.CurrentDomain.FriendlyName);
                        Console.WriteLine("Where:");
                        Console.WriteLine("width:     camera image width to configure. Default is {0}", WIDTH);
                        Console.WriteLine("height:    camera image height to configure. Default is {0}", HEIGHT);
                        Console.WriteLine("numImages: number of images to capture for recording. Default is {0}", NUM_IMAGES);
                        Console.WriteLine("fps:       framerate to use for the recording. Default is {0}\n", FRAMES_PER_SECOND);
                }

                // sets integer value safely
                // (1) ensures increment
                // (2) ensures over minimum
                // (3) ensures below maximum
                // (4) sets value
                static Int64 SetIntValue(ArenaNET.INodeMap nodeMap, String nodeName, Int64 value)
                {
                        // get node
                        ArenaNET.IInteger integer = (ArenaNET.IInteger)nodeMap.GetNode(nodeName);

                        // Ensure increment
                        //    If a node has an increment (all integer nodes & some float
                        //    nodes), only multiples of the increment can be set.
                        value = (((value - integer.Min) / integer.Inc) * integer.Inc) + integer.Min;

        // Check min/max values
        //    Values must not be less than the minimum or exceed the maximum value of
        //    a node. If a value does so, simply push it within range.
        if (value< integer.Min)
                value = integer.Min;

        if (value > integer.Max)
                value = integer.Max;

                        // set value
                        integer.Value = value;

        // return value for output
        return value;
        }

                // sets float value safely
                // (1) ensures over minimum
                // (2) ensures below maximum
                // (3) sets value
                static Double SetFloatValue(ArenaNET.INodeMap nodeMap, String nodeName, Double value)
                {
                        // get node
                        ArenaNET.IFloat floatNode = (ArenaNET.IFloat)nodeMap.GetNode(nodeName);

                        // Check min/max values
                        //    Values must not be less than the minimum or exceed the maximum
                        //    value of a node. If a value does so, simply push it within
                        //    range.
                        if (value < floatNode.Min)
                                value = floatNode.Min;

                        if (value > floatNode.Max)
                                value = floatNode.Max;

                        // set value
                        floatNode.Value = value;

                        // return value for output
                        return value;
                }

                // demonstrates recording a video
                // (1) prepares video parameters
                // (2) prepares video recorder
                // (3) sets video settings
                // (4) opens video
                // (5) appends images
                // (6) closes video
                static void RecordVideo(List<ArenaNET.IImage> images, double fps)
                {
                        // Prepare video parameters
                        Console.WriteLine("{0}Prepare video parameters ({1}x{2}, {3} FPS)", TAB1, images[0].Width, images[0].Height, fps);

                        SaveNET.VideoParams parameters = new SaveNET.VideoParams(
                                images[0].Width,
                                images[0].Height,
                                fps);

                        // Prepare video recorder
                        Console.WriteLine("{0}Prepare video recorder for video {1}", TAB1, FILE_NAME);

                        SaveNET.VideoRecorder recorder = new SaveNET.VideoRecorder(
                                parameters,
                                FILE_NAME);

                        // Set codec, container, and pixel format
                        Console.WriteLine("{0}Set codec to H264, container to MPEG-4, and pixel format to BGR8", TAB1);

                        recorder.SetH264Mp4BGR8();

                        // Open video
                        Console.WriteLine("{0}Open video", TAB1);

                        Console.WriteLine("\nFFMPEG OUTPUT---------------\n");
                        recorder.Open();
                        Console.WriteLine("\nFFMPEG OUTPUT---------------\n");

                        // Append images
                        Console.WriteLine("{0}Append images", TAB2);

                        for (Int32 i = 0; i < images.Count; i++)
                        {
                                if (i % 25 == 0)
                                        Console.Write("{0}", TAB2);
                                Console.Write(".");
                                if (i % images.Count == images.Count - 1)
                                        Console.Write("\n");
                                else if (i % 25 == 24)
                                        Console.Write("\r{0}\r", ERASE_LINE);

                                recorder.AppendImage(images[i].DataArray);
                        }

                        // Close video
                        Console.WriteLine("{0}Close video", TAB1);

                        Console.WriteLine("\nFFMPEG OUTPUT---------------\n");
                        recorder.Close();
                        Console.WriteLine("\nFFMPEG OUTPUT---------------\n");
                }

                // =-=-=-=-=-=-=-=-=-
                // =- PREPARATION -=-
                // =- & CLEAN UP =-=-
                // =-=-=-=-=-=-=-=-=-

                static void Main(string[] args)
                {
                        Console.WriteLine("Cs_Record");

                        // Parse command line args
                        UInt32 width = WIDTH;
                        UInt32 height = HEIGHT;
                        UInt32 numImages = NUM_IMAGES;
                        double fps = FRAMES_PER_SECOND;
                        bool parseResult = true;

                        for (UInt32 i = 0; i < args.Length; i++)
                        {
                                if ((args[i] == "-w") && (i + 1) < args.Length)
                                {
                                        parseResult |= UInt32.TryParse(args[++i], out width);
                                }
                                else if ((args[i] == "-h") && (i + 1) < args.Length)
                                {
                                        parseResult |= UInt32.TryParse(args[++i], out height);
                                }
                                else if ((args[i] == "-n") && (i + 1) < args.Length)
                                {
                                        parseResult |= UInt32.TryParse(args[++i], out numImages);
                                }
                                else if ((args[i] == "-fps") && (i + 1) < args.Length)
                                {
                                        parseResult |= double.TryParse(args[++i], out fps);
                                }
                                else if (args[i] == "--help")
                                {
                                        Usage();
                                        return;
                                }
                                else
                                {
                                        Console.WriteLine("Invalid argument [{0}]", args[i]);
                                        Usage();
                                        return;
                                }
                        }

                        if (height == 0 || width == 0 || !parseResult)
                        {
                                Console.WriteLine("Invalid recording setting specified!");
                                return;
                        }

                        try
                        {
                                // prepare example
                                ArenaNET.ISystem system = ArenaNET.Arena.OpenSystem();
                                system.UpdateDevices(100);
                                if (system.Devices.Count == 0)
                                {
                                        Console.WriteLine("\nNo camera connected\nPress enter to complete");
                                        Console.Read();
                                        return;
                                }
                                else if (numImages == 0)
                                {
                                        Console.WriteLine("\numImages should be greater than 0\nPress enter to complete");
                                        Console.Read();
                                        return;
                                }

                                ArenaNET.IDevice device = system.CreateDevice(system.Devices[0]);

                                // Store initial settings
                                String acquisitionModeInitial;
                                Boolean frameRateEnableInitial = false;
                                double frameRateInitial = 0;
                                Int64 imageWidthInitial = 0;
                                Int64 imageHeightInitial = 0;

                                // Store initial camera settings
                                //    Acquisition mode
                                acquisitionModeInitial = (device.NodeMap.GetNode("AcquisitionMode") as ArenaNET.IEnumeration).Entry.Symbolic;
                                // Framerate enable
                                frameRateEnableInitial = (device.NodeMap.GetNode("AcquisitionFrameRateEnable") as ArenaNET.IBoolean).Value;
                                if (frameRateEnableInitial)
                                {
                                        frameRateInitial = (device.NodeMap.GetNode("AcquisitionFrameRate") as ArenaNET.IFloat).Value;
                                }
                                // Image width
                                imageWidthInitial = (device.NodeMap.GetNode("Width") as ArenaNET.IInteger).Value;
                                // Image height
                                imageHeightInitial = (device.NodeMap.GetNode("Height") as ArenaNET.IInteger).Value;



                                // Set acquisition mode
                                (device.NodeMap.GetNode("AcquisitionMode") as ArenaNET.IEnumeration).FromString("Continuous");

                                // Set width and height
                                //    Reducing the size of an image reduces the amount of
                                //    bandwidth required for each image. The less bandwidth
                                //    required per image, the more images can be sent over the
                                //    same bandwidth.
                                SetIntValue(device.NodeMap, "Width", width);
                                SetIntValue(device.NodeMap, "Height", height);

                                // Set framerate
                                (device.NodeMap.GetNode("AcquisitionFrameRateEnable") as ArenaNET.IBoolean).Value = true;
                                fps = SetFloatValue(device.NodeMap, "AcquisitionFrameRate", fps);

                                Console.Write("Using: \nwidth: {0}\nheight: {1}\nnumImages: {2}\nfps: {3}\n",
                                        (device.NodeMap.GetNode("Width") as ArenaNET.IInteger).Value,
                                        (device.NodeMap.GetNode("Height") as ArenaNET.IInteger).Value,
                                        numImages,
                                        fps);

                                // enable stream auto negotiate packet size
                                var streamAutoNegotiatePacketSizeNode = (ArenaNET.IBoolean)device.TLStreamNodeMap.GetNode("StreamAutoNegotiatePacketSize");
                                streamAutoNegotiatePacketSizeNode.Value = true;

                                // enable stream packet resend
                                var streamPacketResendEnableNode = (ArenaNET.IBoolean)device.TLStreamNodeMap.GetNode("StreamPacketResendEnable");
                                streamPacketResendEnableNode.Value = true;

                                Console.WriteLine("Capturing images");

                                List<ArenaNET.IImage> images = new List<ArenaNET.IImage>();
                                device.StartStream();
                                for (Int32 i = 0; i < numImages; i++)
                                {
                                        if (i % 25 == 0)
                                                Console.Write("{0}", TAB2);
                                        Console.Write(".");
                                        if (i % numImages == numImages - 1)
                                                Console.Write("\n");
                                        else if (i % 25 == 24)
                                                Console.Write("\r{0}\r", ERASE_LINE);

                                        ArenaNET.IImage image = device.GetImage(2000);
                                        images.Add(ArenaNET.ImageFactory.Convert(image, ArenaNET.EPfncFormat.BGR8));
                                        device.RequeueBuffer(image);

                                }
                                device.StopStream();

                                // run example
                                Console.WriteLine("Commence example\n");
                                RecordVideo(images, fps);
                                Console.WriteLine("\nExample complete");

                                // clean up example
                                for (Int32 i = 0; i < images.Count; i++)
                                        ArenaNET.ImageFactory.Destroy(images[i]);

                                // Restore initial settings

                                // Restore width and height
                                SetIntValue(device.NodeMap, "Width", imageWidthInitial);
                                SetIntValue(device.NodeMap, "Height", imageHeightInitial);

                                // Restore acquisition mode
                                (device.NodeMap.GetNode("AcquisitionMode") as ArenaNET.IEnumeration).FromString(acquisitionModeInitial);

                                // Restore framerate
                                (device.NodeMap.GetNode("AcquisitionFrameRateEnable") as ArenaNET.IBoolean).Value = frameRateEnableInitial;

                                if (frameRateEnableInitial)
                                {
                                        SetFloatValue(device.NodeMap, "AcquisitionFrameRate", frameRateInitial);
                                }

                                system.DestroyDevice(device);
                                ArenaNET.Arena.CloseSystem(system);
                        }
                        catch (Exception ex)
                        {
                                Console.WriteLine("\nException thrown: {0}", ex.Message);
                        }

                        Console.WriteLine("Press enter to complete");
                        Console.Read();
                }
        }
}
