Windows Presentation Foundation, Mini Example¶

using System;
using System.Collections.Generic;
using System.Collections.ObjectModel;

namespace MiniExample
{
    // for each node of feature tree
    public class TreeItem
    {
        public TreeItem()
        {
            this.Items = new ObservableCollection<TreeItem>();
        }

        public string Title { get; set; } // string to display
        public string UId { get; set; } // serial number to identify device
        public string Map { get; set; } // which nodemap this node belongs to
        public bool isRoot { get; set; } // true if root
        public bool isLeaf { get; set; } // true if not a category node

        public ObservableCollection<TreeItem> Items { get; set; } // list of children of node
    }

    // for information on selected node
    public class TreeItemInfo
    {
        public string Name { get; set; } // node name
        public string Value { get; set; } // node value
        public string Visible { get; set; } = "Visible"; // "Hidden" for category nodes
        public bool Editable { get; set; } = false; // true when node.AccessMode is RW or WO
        public int Border { get; set; } = 0; // 1 when node value is editable
    }

    public class ArenaManager
    {
        private ArenaNET.ISystem m_system = null;
        private ArenaNET.INodeMap m_nodeMap = null;
        private List<ArenaNET.IDeviceInfo> m_deviceInfos;
        private ArenaNET.IDevice m_connectedDevice = null;
        private ArenaNET.IImage m_converted;

        public ArenaManager()
        {
            m_system = ArenaNET.Arena.OpenSystem();
        }

        // connect to a device based on uid
        public void ConnectDevice(String UId)
        {
            if (m_connectedDevice == null)
            {
                UpdateDevices();

                for (int i = 0; i < m_deviceInfos.Count; i++)
                {
                    if (m_deviceInfos[i].SerialNumber == UId)
                    {
                        m_connectedDevice = m_system.CreateDevice(m_deviceInfos[i]);
                        return;
                    }
                }
            }

            throw new Exception();
        }

        // disconnect from currently connected device
        public void DisconnectDevice()
        {
            m_system.DestroyDevice(m_connectedDevice);
            m_connectedDevice = null;
        }

        // returns if connected to a device
        public bool DeviceConnected()
        {
            return (m_connectedDevice != null);
        }

        // gets a specific nodemap
        public void GetNodeMap(String nodeMap = "DeviceNodeMap")
        {
            if (m_connectedDevice == null)
                throw new Exception();

            if (nodeMap == "DeviceNodeMap")
                m_nodeMap = m_connectedDevice.NodeMap;
            else if (nodeMap == "TLDeviceNodeMap")
                m_nodeMap = m_connectedDevice.TLDeviceNodeMap;
            else if (nodeMap == "TLInterfaceNodeMap")
                m_nodeMap = m_connectedDevice.TLInterfaceNodeMap;
            else if (nodeMap == "TLStreamNodeMap")
                m_nodeMap = m_connectedDevice.TLStreamNodeMap;
            else if (nodeMap == "TLSystemNodeMap")
                m_nodeMap = m_system.TLSystemNodeMap;
            else
                throw new Exception();
        }

        // gets a list of uids of available devices
        public List<String> GetDeviceUIds()
        {
            UpdateDevices();

            List<String> uids = new List<String>();

            for (int i = 0; i < m_deviceInfos.Count; i++)
            {
                uids.Add(m_deviceInfos[i].SerialNumber);
            }

            return uids;
        }

        // gets a list of names of child nodes of a specific node
        public List<String> GetNodeChildren(String nodeName)
        {
            if (m_nodeMap == null)
                throw new Exception();

            List<String> childNodes = new List<string>();

            ArenaNET.INode node = m_nodeMap.GetNode(nodeName);
            ArenaNET.ICategory category = node as ArenaNET.ICategory;
            List<ArenaNET.IValue> children = category.Features;
            foreach (ArenaNET.IValue val in children)
            {
                childNodes.Add(val.Name);
            }

            return childNodes;
        }

        // gets uid of currently connected device
        public String ConnectedDeviceUId()
        {
            return ((ArenaNET.IString)m_connectedDevice.NodeMap.GetNode("DeviceSerialNumber")).Value;
        }

        // gets name of device
        public String GetDeviceName(String UId, String node)
        {
            for (int i = 0; i < m_deviceInfos.Count; i++)
            {
                if (m_deviceInfos[i].SerialNumber == UId)
                {
                    if (node == "DeviceUserID" || node == "UserDefinedName")
                        return m_deviceInfos[i].UserDefinedName;
                    else if (node == "DeviceModelName" || node == "ModelName")
                        return m_deviceInfos[i].ModelName;
                }
            }

            return "Invalid argument";
        }

        // gets access mode of a node
        public String GetNodeAccessMode(String nodeName)
        {
            if (m_nodeMap == null)
                throw new Exception();

            ArenaNET.INode node = m_nodeMap.GetNode(nodeName);

            return node.AccessMode.ToString();
        }

        // gets name of a node
        public String GetNodeName(String nodeName)
        {
            if (m_nodeMap == null)
                throw new Exception();

            ArenaNET.INode node = m_nodeMap.GetNode(nodeName);

            return node.Name;
        }

        // gets interface type of a node
        public String GetNodeInterfaceType(String nodeName)
        {
            if (m_nodeMap == null)
                throw new Exception();


            ArenaNET.INode node = m_nodeMap.GetNode(nodeName);

            return node.PrincipalInterfaceType.ToString();
        }

        // gets value of a node
        public String GetNodeValue(String nodeName)
        {
            if (m_nodeMap == null)
                throw new Exception();

            ArenaNET.INode node = m_nodeMap.GetNode(nodeName);
            String interfaceType = node.PrincipalInterfaceType.ToString();
            String ret = null;

            if (interfaceType == "intfIBoolean")
            {
                ArenaNET.IBoolean iNode = (ArenaNET.IBoolean)node;
                ret = iNode.Value.ToString();
            }
            else if (interfaceType == "intfICommand")
            {
                // ICommand node has no stored value, set default to 0 as inputting a 1 means execute
                ArenaNET.ICommand iNode = (ArenaNET.ICommand)node;
                ret = "0";
            }
            else if (interfaceType == "intfIEnumEntry")
            {
                ArenaNET.IEnumEntry iNode = (ArenaNET.IEnumEntry)node;
                ret = iNode.ToString();
            }
            else if (interfaceType == "intfIEnumeration")
            {
                ArenaNET.IEnumeration iNode = (ArenaNET.IEnumeration)node;
                ret = iNode.ToString();
            }
            else if (interfaceType == "intfIFloat")
            {
                ArenaNET.IFloat iNode = (ArenaNET.IFloat)node;
                ret = Math.Round(iNode.Value, 1).ToString();
            }
            else if (interfaceType == "intfIInteger")
            {
                ArenaNET.IInteger iNode = (ArenaNET.IInteger)node;
                ret = iNode.Value.ToString();
            }
            else if (interfaceType == "intfIRegister")
            {
                ArenaNET.IRegister iNode = (ArenaNET.IRegister)node;
                ret = iNode.ToString();
            }
            else if (interfaceType == "intfIString")
            {
                ArenaNET.IString iNode = (ArenaNET.IString)node;
                ret = iNode.Value;
            }
            else
            {
                ret = "Error getting value of node with interface type " + interfaceType;
            }

            return ret;
        }

        // gets additional info of a node
        public String GetAdditionalNodeInfo(String nodeName)
        {
            if (m_nodeMap == null)
                throw new Exception();

            ArenaNET.INode node = m_nodeMap.GetNode(nodeName);
            String interfaceType = node.PrincipalInterfaceType.ToString();
            String ret = "";

            if (interfaceType == "intfICommand")
            {
                ArenaNET.ICommand iNode = (ArenaNET.ICommand)node;

                ret = "(To execute this command, enter '1' in text box)";
            }
            else if (interfaceType == "intfIEnumeration")
            {
                ArenaNET.IEnumeration iNode = (ArenaNET.IEnumeration)node;

                List<String> entries = iNode.Symbolics;
                ret = "- ";

                foreach (var entry in entries)
                {
                    if (ret == "- ")
                        ret += entry;
                    else
                        ret += "\n- " + entry;
                }
            }
            else if (interfaceType == "intfIFloat")
            {
                ArenaNET.IFloat iNode = (ArenaNET.IFloat)node;

                ret = "  Maximum: " + iNode.Max.ToString()
                    + "\n  Minimum: " + iNode.Min.ToString();
            }
            else if (interfaceType == "intfIInteger")
            {
                ArenaNET.IInteger iNode = (ArenaNET.IInteger)node;

                ret = "  Maximum: " + iNode.Max.ToString()
                    + "\n  Minimum: " + iNode.Min.ToString()
                    + "\n  Increment: " + iNode.Inc.ToString();
            }
            else
            {
                throw new Exception();
            }

            return ret;
        }

        // returns if node is readable
        public bool IsNodeReadable(String nodeName)
        {
            if (m_nodeMap == null)
                throw new Exception();

            ArenaNET.INode node = m_nodeMap.GetNode(nodeName);

            return node.IsReadable;
        }

        // returns if node is read only (not writable)
        public bool IsNodeReadOnly(String nodeName)
        {
            if (m_nodeMap == null)
                throw new Exception();

            return GetNodeAccessMode(nodeName) == "RO";
        }

        // returns if node is a specific type
        public bool IsNodeType(String nodeName, String nodeType)
        {
            if (m_nodeMap == null)
                throw new Exception();

            ArenaNET.INode node = m_nodeMap.GetNode(nodeName);
            String interfaceType = node.PrincipalInterfaceType.ToString();

            return interfaceType == nodeType;
        }

        // check if ip addresses between device and interface match based on their subnets
        public bool IsNetworkValid(String UId)
        {
            UpdateDevices();

            for (int i = 0; i < m_deviceInfos.Count; i++)
            {
                if (m_deviceInfos[i].SerialNumber == UId)
                {
                    UInt32 ip = (UInt32)m_deviceInfos[i].IpAddress;
                    UInt32 subnet = (UInt32)m_deviceInfos[i].SubnetMask;

                    ArenaNET.IInteger ifipNode = (ArenaNET.IInteger)m_system
                            .GetTLInterfaceNodeMap(m_deviceInfos[i]).GetNode("GevInterfaceSubnetIPAddress");
                    ArenaNET.IInteger ifsubnetNode = (ArenaNET.IInteger)m_system
                            .GetTLInterfaceNodeMap(m_deviceInfos[i]).GetNode("GevInterfaceSubnetMask");
                    UInt32 ifip = (UInt32)ifipNode.Value;
                    UInt32 ifsubnet = (UInt32)ifsubnetNode.Value;

                    if (subnet != ifsubnet)
                        return false;

                    if ((ip & subnet) != (ifip & ifsubnet))
                        return false;

                    return true;
                }
            }

            throw new Exception();
        }

        // get image as a bitmap
        public System.Drawing.Bitmap GetImage(UInt32 timeout = 2000)
        {
            if (m_converted != null)
            {
                ArenaNET.ImageFactory.Destroy(m_converted);
                m_converted = null;
            }

            ArenaNET.IImage image = m_connectedDevice.GetImage(timeout);
            m_converted = ArenaNET.ImageFactory.Convert(image, (ArenaNET.EPfncFormat)0x02200017);
            m_connectedDevice.RequeueBuffer(image);

            return m_converted.Bitmap;
        }

        // start stream
        public void StartStream()
        {
            if (m_connectedDevice != null)
            {
                if ((m_connectedDevice.TLStreamNodeMap.GetNode("StreamIsGrabbing") as ArenaNET.IBoolean).Value == false)
                {
                    (m_connectedDevice.TLStreamNodeMap.GetNode("StreamBufferHandlingMode") as ArenaNET.IEnumeration).Symbolic = "NewestOnly";
                    m_connectedDevice.StartStream();
                }
            }
        }

        // stop stream
        public void StopStream()
        {
            if (m_connectedDevice != null)
            {
                if ((m_connectedDevice.TLStreamNodeMap.GetNode("StreamIsGrabbing") as ArenaNET.IBoolean).Value == true)
                {
                    m_connectedDevice.StopStream();
                }
            }
        }

        // returns if connected device is streaming
        public bool IsStreaming()
        {
            return (m_connectedDevice.TLStreamNodeMap.GetNode("StreamIsGrabbing") as ArenaNET.IBoolean).Value;
        }

        // sets node to a value
        public void SetNodeValue(String nodeName, String val)
        {
            if (m_nodeMap == null)
                throw new Exception();

            ArenaNET.INode node = m_nodeMap.GetNode(nodeName);

            String interfaceType = node.PrincipalInterfaceType.ToString();

            if (interfaceType == "intfIBoolean")
            {
                ArenaNET.IBoolean iNode = (ArenaNET.IBoolean)node;
                iNode.Value = Convert.ToBoolean(val);
            }
            else if (interfaceType == "intfICommand")
            {
                ArenaNET.ICommand iNode = (ArenaNET.ICommand)node;
                if (val == "1")
                    iNode.Execute();
                else
                    throw new Exception();
            }
            else if (interfaceType == "intfIEnumEntry")
            {
                ArenaNET.IEnumEntry iNode = (ArenaNET.IEnumEntry)node;
                iNode.FromString(val);
            }
            else if (interfaceType == "intfIEnumeration")
            {
                ArenaNET.IEnumeration iNode = (ArenaNET.IEnumeration)node;
                iNode.FromString(val);
            }
            else if (interfaceType == "intfIFloat")
            {
                ArenaNET.IFloat iNode = (ArenaNET.IFloat)node;
                iNode.Value = (float)Convert.ToInt32(val);
            }
            else if (interfaceType == "intfIInteger")
            {
                ArenaNET.IInteger iNode = (ArenaNET.IInteger)node;
                iNode.Value = Convert.ToInt32(val);
            }
            else if (interfaceType == "intfIRegister")
            {
                ArenaNET.IRegister iNode = (ArenaNET.IRegister)node;
                iNode.FromString(val);
            }
            else if (interfaceType == "intfIString")
            {
                ArenaNET.IString iNode = (ArenaNET.IString)node;
                iNode.Value = val;
            }
        }

        // update list of available devices
        private void UpdateDevices()
        {
            m_system.UpdateDevices(100);
            m_deviceInfos = m_system.Devices;
        }
    }
}
