using System;

namespace Cs_Sequencer_HDR
{
        class Cs_Sequencer_HDR
        {
                const String TAB1 = "  ";
                const String TAB2 = "    ";
                const String TAB3 = "      ";

                // Sequencer: HDR
                //    This example demonstrates saving a set of images using the
                //    sequencer. It will cycle through 3 sequencer states each one using
                //    a different exposure time. The images could then be used to
                //    generate an HDR image. Much like a state machine, in order to use
                //    the sequencer we must initialize each set appropriately. Each set
                //    can have its own exposure time and gain, and will contain
                //    information such as the sequencer starting position as well as
                //    paths to other sets. A set can have multiple paths where each path
                //    has its own next set, trigger source and trigger activation. In
                //    this example the sequencer has 3 sets where set 0 goes to set 1,
                //    set 1 goes to set 2 and set 2 goes back to set 0, all being
                //    triggered on Frame Start.

                // =-=-=-=-=-=-=-=-=-
                // =-=- SETTINGS =-=-
                // =-=-=-=-=-=-=-=-=-

                // Exposure time
                //    Time that the sensor will be exposed when grabbing an image (in
                //    microseconds). Exposure time depends on the application, but
                //    generally the less available the light, the higher the exposure
                //    time. Storing multiple values so that each consecutive state can be
                //    saved with varying exposure times.
                static double EXPOSURE_TIME_0 = 25000.0;
                static double EXPOSURE_TIME_1 = 50000.0;
                static double EXPOSURE_TIME_2 = 100000.0;

                // timeout for grabbing an image (in milliseconds)
                const UInt32 IMAGE_TIMEOUT = 2000;

                // number of states in sequencer (<= max range of SequencerSetSelector
                // feature)
                const UInt32 NUM_SETS = 3;

                // name of file to save
                const String FILE_NAME_PATTERN = "Images/Cs_Sequencer_HDR_<count>.jpg";

                // pixel format
                const ArenaNET.EPfncFormat PIXEL_FORMAT = ArenaNET.EPfncFormat.BGR8;

                // =-=-=-=-=-=-=-=-=-
                // =-=- EXAMPLE -=-=-
                // =-=-=-=-=-=-=-=-=-

                static void SetSequencerSet(ArenaNET.IDevice device, UInt32 setNumber, double exposureTime, UInt32 pathNextSet, string triggerSource)
                {
                        // set Sequencer Set Selector to sequence number
                        var sequencerSetSelectorNode = device.NodeMap.GetNode("SequencerSetSelector") as ArenaNET.IInteger;
                        sequencerSetSelectorNode.Value = setNumber;

                        Console.WriteLine("{0}Updating set {1}", TAB2, setNumber);

                        var sequencerFeatureSelectorNode = device.NodeMap.GetNode("SequencerFeatureSelector") as ArenaNET.IEnumeration;
                        sequencerFeatureSelectorNode.FromString("ExposureTime");

                        // set exposure time to the value we want it to save
                        var exposureTimeNode = device.NodeMap.GetNode("ExposureTime") as ArenaNET.IFloat;
                        exposureTimeNode.Value = exposureTime;

                        Console.WriteLine("{0}Exposure time = {1}", TAB3, exposureTime);

                        // select the path we want it to follow from this set to next set,
                        // there can be multiple paths so the first path will always be set
                        // to 0
                        var sequencerPathSelectorNode = device.NodeMap.GetNode("SequencerPathSelector") as ArenaNET.IInteger;
                        sequencerPathSelectorNode.Value = 0;

                        // set next state in the sequence, ensure it does not exceed the
                        // maximum
                        var sequencerSetNextNode = device.NodeMap.GetNode("SequencerSetNext") as ArenaNET.IInteger;
                        sequencerSetNextNode.Value = pathNextSet;

                        Console.WriteLine("{0}Path[0] = {1}", TAB3, pathNextSet);

                        // set Sequencer Trigger Source to Frame Start
                        var sequencerTriggerSourceNode = device.NodeMap.GetNode("SequencerTriggerSource") as ArenaNET.IEnumeration;
                        sequencerTriggerSourceNode.FromString(triggerSource);

                        Console.WriteLine("{0}Trigger source = {1}", TAB3, triggerSource);

                        // Save current state
                        //    Once all appropriate settings have been configured, make sure
                        //    to save the state to the sequence. Notice that these settings
                        //    will be lost when the camera is power-cycled.

                        Console.WriteLine("{0}Save sequence set", TAB3);

                        var pSequencerSetSave = (ArenaNET.ICommand)device.NodeMap.GetNode("SequencerSetSave");
                        pSequencerSetSave.Execute();
                }

                // start streaming, acquire and save images
                static void AcquireAndSaveImages(ArenaNET.IDevice device, String filename)
                {
                        // enable stream auto negotiate packet size
                        var streamAutoNegotiatePacketSizeNode = (ArenaNET.IBoolean)device.TLStreamNodeMap.GetNode("StreamAutoNegotiatePacketSize");
                        streamAutoNegotiatePacketSizeNode.Value = true;

                        // enable stream packet resend
                        var streamPacketResendEnableNode = (ArenaNET.IBoolean)device.TLStreamNodeMap.GetNode("StreamPacketResendEnable");
                        streamPacketResendEnableNode.Value = true;

                        Console.WriteLine("{0}Start streaming", TAB2);
                        device.StartStream();

                        // set an image in each set of sequencer
                        Console.WriteLine("{0}Getting {1} images", TAB2, NUM_SETS);

                        // create empty parameters so we can prepare writer before loop
                        SaveNET.ImageParams emptyParameters = new SaveNET.ImageParams();

                        // prepare image writer
                        SaveNET.ImageWriter writer = new SaveNET.ImageWriter(emptyParameters, filename);

                        for (int i = 0; i < NUM_SETS; i++)
                        {
                                // set image
                                ArenaNET.IImage image = device.GetImage(IMAGE_TIMEOUT);

                                // prepare image parameters from device settings
                                ArenaNET.IImage converted = ArenaNET.ImageFactory.Convert(image, PIXEL_FORMAT);

                                // update image writer to use appropriate image parameters
                                SaveNET.ImageParams parameters = new SaveNET.ImageParams(converted.Width, converted.Height, converted.BitsPerPixel, true);
                                writer.Params = parameters;

                                // save image
                                writer.Save(converted.DataArray, true);

                                Console.WriteLine("{0}Converted and saved image {1}", TAB2, writer.GetLastFileName());

                                // destroy converted image
                                ArenaNET.ImageFactory.Destroy(converted);

                                // requeue image buffer
                                device.RequeueBuffer(image);
                        }

                        // stop stream
                        Console.WriteLine("{0}Stop streaming", TAB2);
                        device.StopStream();
                }

                static void AcquireImagesUsingSequencer(ArenaNET.IDevice device)
                {
                        // get node values that will be changed in order to return their
                        // values at the end of the example
                        var exposureAutoNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("ExposureAuto");
                        String exposureAutoInitial = exposureAutoNode.Entry.Symbolic;

                        var gainAutoNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("GainAuto");
                        String gainAutoInitial = gainAutoNode.Entry.Symbolic;

                        var exposureTimeNode = (ArenaNET.IFloat)device.NodeMap.GetNode("ExposureTime");
                        double exposureTimeInitial = exposureTimeNode.Value;

                        // Disable automatic exposure and gain
                        //    Disable automatic exposure  and gain before setting an exposure
                        //    time. Automatic exposure and gain controls whether they are set
                        //    manually or automatically by the device. Setting automatic
                        //    exposure and gain to 'Off' stops the device from automatically
                        //    updating the exposure time while streaming.
                        Console.WriteLine("{0}Disable automatic exposure and gain", TAB1);

                        exposureAutoNode.FromString("Off");

                        gainAutoNode.FromString("Off");

                        // if sequencer mode is on, turn it off
                        var sequencerModeNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("SequencerMode");
                        String sequencerModeInitial = gainAutoNode.Entry.Symbolic;
                        if (sequencerModeInitial == "On")
                        {
                                Console.WriteLine("{0}Turn sequencer mode off", TAB1);
                                sequencerModeNode.FromString("Off");
                        }

                        // Put sequencer in configuration mode
                        //    Sequencer configuration mode must be on while making changes to
                        //    the sequencer sets.
                        Console.WriteLine("{0}Turn sequencer configuration mode on", TAB1);
                        var sequencerConfigurationModeNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("SequencerConfigurationMode");
                        sequencerConfigurationModeNode.FromString("On");

                        // get maximum exposure time to ensure set of exposure times are all
                        // less than max
                        double exposureTimeMin = exposureTimeNode.Min;
                        double exposureTimeMax = exposureTimeNode.Max;

                        // if largest exposure times is greater than max exposure time, set
                        // largest exposure time to max value and set the remaining exposure
                        // times to half the value of the state before
                        if (EXPOSURE_TIME_2 > exposureTimeMax || EXPOSURE_TIME_0 < exposureTimeMin )
                        {
                                EXPOSURE_TIME_2 = exposureTimeMax;
                                EXPOSURE_TIME_1 = EXPOSURE_TIME_2 / 2;
                                EXPOSURE_TIME_0 = EXPOSURE_TIME_1 / 2;
                        }

                        // set up sequencer sets
                        Console.WriteLine("{0}Set up sequencer sets", TAB1);

                        SetSequencerSet(device, 0, EXPOSURE_TIME_0, 1, "FrameStart");
                        SetSequencerSet(device, 1, EXPOSURE_TIME_1, 2, "FrameStart");
                        SetSequencerSet(device, 2, EXPOSURE_TIME_2, 0, "FrameStart");

                        // sets the sequencer starting set to be set 0
                        var sequencerSetStartNode = device.NodeMap.GetNode("SequencerSetStart") as ArenaNET.IInteger;
                        sequencerSetStartNode.Value = 0;

                        // turn off configuration mode
                        Console.WriteLine("{0}Turn sequencer configuration mode off", TAB1);
                        sequencerConfigurationModeNode.FromString("Off");

                        // Turn on sequencer
                        //    When sequencer mode is on and the device is streaming it will
                        //    follow the sequencer sets according to their saved settings.
                        Console.WriteLine("{0}Turn sequencer mode on", TAB1);
                        sequencerModeNode.FromString("On");

                        // Acquire and Save images
                        //    This function will start the stream, acquire an image in each
                        //    set of the sequencer using its corresponding settings, save
                        //    each image and then stop the stream.
                        AcquireAndSaveImages(device, FILE_NAME_PATTERN);

                        // turn off sequencer mode
                        Console.WriteLine("{0}Turn sequencer mode off", TAB1);
                        sequencerModeNode.FromString("Off");

                        // return nodes to their initial values
                        Console.WriteLine("{0}Returning nodes to their initial values", TAB1);
                        exposureTimeNode.Value = exposureTimeInitial;
                        exposureAutoNode.FromString(exposureAutoInitial);
                        gainAutoNode.FromString(gainAutoInitial);
                }

                        // =-=-=-=-=-=-=-=-=-
                        // =- PREPARATION -=-
                        // =- & CLEAN UP =-=-
                        // =-=-=-=-=-=-=-=-=-

                        static void Main(string[] args)
                {
                        Console.WriteLine("Cs_Sequencer_HDR");

                        try
                        {
                                // prepare example
                                ArenaNET.ISystem system = ArenaNET.Arena.OpenSystem();
                                system.UpdateDevices(100);
                                if (system.Devices.Count == 0)
                                {
                                        Console.WriteLine("\nNo camera connected\nPress enter to complete");
                                        Console.Read();
                                        return;
                                }
                                ArenaNET.IDevice device = system.CreateDevice(system.Devices[0]);

                                // run example
                                Console.WriteLine("Commence example\n");
                                AcquireImagesUsingSequencer(device);
                                Console.WriteLine("\nExample complete");

                                // clean up example
                                system.DestroyDevice(device);
                                ArenaNET.Arena.CloseSystem(system);
                        }
                        catch (Exception ex)
                        {
                                Console.WriteLine("Exception thrown: {0}", ex.Message);
                        }

                        Console.WriteLine("Press enter to complete");
                        Console.Read();
                }
        }
}
