using System;

namespace Cs_PixelCorrection
{
        class Cs_PixelCorrection
        {
                const String TAB1 = "  ";
                const String TAB2 = "    ";
                const String TAB3 = "      ";

                // Pixel Correction
                //    This example introduces the basics of pixel correction. A single
                //    arbitrary pixel is chosen and added to the device's pixel
                //    correction list. These changes are then saved to the camera before
                //    being removed.

                // =-=-=-=-=-=-=-=-=-
                // =-=- SETTINGS =-=-
                // =-=-=-=-=-=-=-=-=-

                // Pixel values
                //    This example does not search for pixels to correct, but instead
                //    just uses arbitrary pixels to demonstrate what would be done if if
                //    a bad pixel were found.
                const Int64 PIXEL_X = 256;
                const Int64 PIXEL_Y = 128;


                // =-=-=-=-=-=-=-=-=-
                // =-=- EXAMPLE -=-=-
                // =-=-=-=-=-=-=-=-=-

                // demonstrates pixel correction
                // (1) enables pixel correction
                // (2) adds new pixel to correction list
                // (3) sets new pixel coordinates
                // (4) applies changes to correction list
                // (5) saves correction to camera
                // (6) removes pixel from correction list
                static void CorrectPixels(ArenaNET.IDevice device, Int64 pixelX, Int64 pixelY)
                {
                        // get node values that will be changed in order to return their
                        // values at the end of the example
                        var pixelCorrectionEnableNode = (ArenaNET.IBoolean)device.NodeMap.GetNode("DefectCorrectionEnable");
                        Boolean pixelCorrectionEnableInitial = pixelCorrectionEnableNode.Value;

                        // Enable pixel correction
                        Console.WriteLine("{0}Enable pixel correction", TAB1);

                        pixelCorrectionEnableNode.Value = true;

                        // get initial correction count
                        var pixelCorrectionCountNode = (ArenaNET.IInteger)device.NodeMap.GetNode("DefectCorrectionCount");
                        Int64 pixelCorrectionCountInitial = pixelCorrectionCountNode.Value;

                        // Add new pixel to end of correction list.
                        //    Getting a new defect automatically updates the pixel correction
                        //    index.
                        Console.WriteLine("{0}Add pixel to correction list", TAB1);

                        var pixelCorrectionNewDefectNode = (ArenaNET.ICommand)device.NodeMap.GetNode("DefectCorrectionGetNewDefect");
                        pixelCorrectionNewDefectNode.Execute();

                        // get updated pixel index
                        var pixelCorrectionIndexNode = (ArenaNET.IInteger)device.NodeMap.GetNode("DefectCorrectionIndex");
                        var pixelCorrectionUpdatedIndex = pixelCorrectionIndexNode.Value;

                        Console.Write("{0}Pixel index: {1} ", TAB2, pixelCorrectionUpdatedIndex.ToString().PadLeft(2));

                        // Set pixel position
                        var pixelCorrectionPositionXNode = (ArenaNET.IInteger)device.NodeMap.GetNode("DefectCorrectionPositionX");
                        pixelCorrectionPositionXNode.Value = pixelX;

                        Console.Write("(x: {0}", pixelCorrectionPositionXNode.Value.ToString().PadLeft(4));

                        var pixelCorrectionPositionYNode = (ArenaNET.IInteger)device.NodeMap.GetNode("DefectCorrectionPositionY");
                        pixelCorrectionPositionYNode.Value = pixelY;

                        Console.WriteLine(", y: {0})", pixelCorrectionPositionYNode.Value.ToString().PadLeft(4));

                        // Apply correction list
                        //    Once the pixel corrections are applied, they will take effect
                        //    immediately. However, by power-cycling the camera, the defect
                        //    list will go back to its default values.
                        Console.WriteLine("{0}Apply correction", TAB1);

                        var pixelCorrectionApplyNode = (ArenaNET.ICommand)device.NodeMap.GetNode("DefectCorrectionApply");
                        pixelCorrectionApplyNode.Execute();

                        // Save correction to camera
                        //    Optionally write the correction to the camera to make the
                        //    changes permanent (the camera can still be set to default by
                        //    executing the DefectCorrectionRestoreDefault node)
                        Console.WriteLine("{0}Save correction to camera", TAB1);

                        var pixelCorrectionSaveNode = (ArenaNET.ICommand)device.NodeMap.GetNode("DefectCorrectionSave");
                        pixelCorrectionSaveNode.Execute();

                        // Remove all pixels set through this example by decrementing through the index
                        //    The index is updated so that there are no empty indices after
                        //    removal, regardless of position
                        Console.WriteLine("{0}Find and remove pixel from correction list", TAB1);

                        for (Int64 pixelCorrectionIndex = pixelCorrectionUpdatedIndex; pixelCorrectionIndex >= pixelCorrectionCountInitial; pixelCorrectionIndex--)
                        {
                                // select pixel
                                pixelCorrectionIndexNode.Value = pixelCorrectionIndex;

                                Console.Write("{0}Pixel index: {1} ", TAB2, pixelCorrectionIndex.ToString().PadLeft(2));

                                // check position
                                Int64 x = pixelCorrectionPositionXNode.Value;

                                Console.Write("(x: {0}", x.ToString().PadLeft(4));

                                Int64 y = pixelCorrectionPositionYNode.Value;

                                Console.Write(", y: {0})", y.ToString().PadLeft(4));

                                if (x == pixelX && y == pixelY)
                                {
                                        Console.WriteLine(" matches\n{0}Remove pixel", TAB2);

                                        // Delete pixel from correction list
                                        var pixelCorrectionRemoveNode = (ArenaNET.ICommand)device.NodeMap.GetNode("DefectCorrectionRemove");
                                        pixelCorrectionRemoveNode.Execute();
                                        break;
                                }
                                else
                                        Console.WriteLine(" does not match");
                        }


                        // return nodes to their initial values
                        pixelCorrectionEnableNode.Value = pixelCorrectionEnableInitial;
                }

                // =-=-=-=-=-=-=-=-=-
                // =- PREPARATION -=-
                // =- & CLEAN UP =-=-
                // =-=-=-=-=-=-=-=-=-

                static void Main(string[] args)
                {
                        Console.WriteLine("Cs_PixelCorrection");

                        try
                        {
                                // prepare example
                                ArenaNET.ISystem system = ArenaNET.Arena.OpenSystem();
                                system.UpdateDevices(100);
                                if (system.Devices.Count == 0)
                                {
                                        Console.WriteLine("\nNo camera connected\nPress enter to complete");
                                        Console.Read();
                                        return;
                                }
                                ArenaNET.IDevice device = system.CreateDevice(system.Devices[0]);

                                var pixelCorrectionNode = (ArenaNET.IInteger)device.NodeMap.GetNode("DefectCorrectionCount");
                                Int64 pixelCorrectionCount = pixelCorrectionNode.Value;

                                // ensure arbitrary pixel selections are not already corrected
                                if (pixelCorrectionCount > 0)
                                {
                                        for (Int64 i = 0; i < pixelCorrectionCount; i++)
                                        {
                                                var pixelCorrectionIndexNode = (ArenaNET.IInteger)device.NodeMap.GetNode("DefectCorrectionIndex");
                                                pixelCorrectionIndexNode.Value = i;

                                                var pixelCorrectionPositionXNode = (ArenaNET.IInteger)device.NodeMap.GetNode("DefectCorrectionPositionX");
                                                var pixelCorrectionPositionYNode = (ArenaNET.IInteger)device.NodeMap.GetNode("DefectCorrectionPositionY");

                                                if (pixelCorrectionPositionXNode.Value == PIXEL_X && pixelCorrectionPositionYNode.Value == PIXEL_Y)
                                                {
                                                        Console.WriteLine("\nPixels already corrected\nPress enter to complete");
                                                        Console.Read();
                                                        return;
                                                }
                                        }
                                }

                                // run example
                                Console.WriteLine("Commence example\n");
                                CorrectPixels(device, PIXEL_X, PIXEL_Y);
                                Console.WriteLine("\nExample complete");

                                // clean up example
                                system.DestroyDevice(device);
                                ArenaNET.Arena.CloseSystem(system);
                        }
                        catch (Exception ex)
                        {
                                Console.WriteLine("Exception thrown: {0}", ex.Message);
                        }

                        Console.WriteLine("Press enter to complete");
                        Console.Read();
                }
        }
}
