using System;
using System.Collections.Generic; // For List<>

namespace Cs_Explore_NodeTypes
{
        class Cs_Explore_NodeTypes
        {
                const String TAB1 = "  ";
                const String TAB2 = "    ";
                const String TAB3 = "      ";

                // Explore: Node Types
                //    This example explores the different properties of various node's
                //    types including boolean, string, enumeration, integer, and float
                //    nodes. The user inputs the node name that they wish to access
                //    (leaving out spacing between words) in order to retrieve the node
                //    properties, or inputs 'x' to exit. This example includes commented
                //    out code that sets values for each node type, retrieves entry by
                //    node name for enumeration types, and imposes maximum and minimum
                //    values for integer and float types. See Cpp_Explore_Nodes for a
                //    complete list of nodes and their respective types.

                // =-=-=-=-=-=-=-=-=-
                // =-=- EXAMPLE -=-=-
                // =-=-=-=-=-=-=-=-=-

                // explores nodes of boolean type
                // (1) retrieves value
                // (2) demonstrates value setter in a comment
                static void ExploreBoolean(ArenaNET.IBoolean boolean)
                {
                        // Retrieve value
                        Boolean value = boolean.Value;

                        Console.WriteLine("{0}Value: {1}", TAB3, value);

                        // Set value
                        //    {
                        // boolean.Value = true;
                        //    }
                }

                // explores nodes of string type
                // (1) retrieves value
                // (2) retrieves maximum value length
                // (3) demonstrates value setter in a comment
                static void ExploreString(ArenaNET.IString str)
                {
                        // Retrieve value
                        String value = str.Value;

                        Console.WriteLine("{0}Value: {1}", TAB3, value);

                        // Retrieve maximum length
                        long length = str.MaxLength;

                        Console.WriteLine("{0}Maximum length: {1}", TAB3, length);

                        // Set value
                        //    {
                        // str.Value = "string value";
                        //    }
                }

                // explores nodes of type enumeration
                // (1) retrieves currently set enum entry node
                // (2) retrieves value
                // (3) retrieves list of entries
                // (4) demonstrates list of symbolics retrieval in a comment
                // (5) demonstrates enum entry node retrieval by name in a comment
                // (6) demonstrates value setter in a comment
                static void ExploreEnumeration(ArenaNET.IEnumeration enumeration)
                {
                        // Retrieve current entry
                        String entrySymbolic = enumeration.Symbolic;

                        Console.WriteLine("{0}Current entry: {1}", TAB3, entrySymbolic);

                        // Retrieve integer value
                        long intValue = enumeration.IntValue;

                        Console.WriteLine("{0}Integer value: {1}", TAB3, intValue);

                        // Retrieve entries
                        List<ArenaNET.IEnumEntry> entries = enumeration.Entries;

                        Console.Write("{0}Entries: ", TAB3);

                        for(int i = 0; i < entries.Count; i++)
                        {
                                if(i > 0)
                                        Console.Write(", ");

                                ArenaNET.IEnumEntry entry = entries[i];
                                Console.Write("{0}", entry.Symbolic);

                        }

                        Console.Write("\n");

                        // Retrieve symbolics
                        //    {
                        //        List<String> list = enumeration.Symbolics;
                        //    }

                        // Retrieve entry by name
                        //    {
                        //        String symbolic = "symbolic value";
                        //    ArenaNET.IEnumEntry entry =
                        //    enumeration.GetEntryByName(symbolic);
                        //    }

                        // Set value
                        //    {
                        //         String symbolic = "symbolic value";
                        //    ArenaNET.IEnumEntry entry =
                        //    enumeration.GetEntryByName(symbolic); long intValue =
                        //    entry.IntValue; enumeration.IntValue = intValue;
                        //    }
                }

                // explores nodes of type integer
                // (1) retrieves value
                // (2) retrieves maximum and minimum
                // (3) retrieves increment and increment mode
                // (4) retrieves representation
                // (5) retrieves unit
                // (6) demonstrates maximum and minimum imposition in a comment
                // (7) demonstrates value setter in a comment
                static void ExploreInteger(ArenaNET.IInteger integer)
                {
                        // Retrieve value
                        long value = integer.Value;

                        Console.WriteLine("{0}Value: {1}", TAB3, value);

                        // Retrieve range
                        long max = integer.Max;
                        long min = integer.Min;

                        Console.WriteLine("{0}Minimum, maximum,: {1}, {2}", TAB3, min, max);

                        // Retrieve increment
                        long inc = integer.Inc;
                        ArenaNET.EIncMode incMode = integer.IncMode;
                        String incModeStr = incMode.ToString();

                        Console.WriteLine("{0}Increment (mode): {1} ({2})", TAB3, inc, incModeStr);

                        // Retrieve representation
                        ArenaNET.ERepresentation rep = integer.Representation;
                        String repStr = rep.ToString();

                        Console.WriteLine("{0}Representation: {1}", TAB3, repStr);

                        // Retrieve unit
                        String unit = integer.Unit;

                        Console.WriteLine("{0}Unit: {1}", TAB3, unit);

                        // Impose maximum and minimum
                        //    {
                        // long newMax = 10; long newMin = 0; integer.ImposeMax(newMax);
                        // integer.ImposeMin(newMin);
                        //    }

                        // Set value
                        //    {
                        // long value = 0; integer.Value = value;
                        //    }
                }

                // explores nodes of type float
                // (1) retrieves value
                // (2) retrieves maximum and minimum
                // (3) retrieves increment and increment mode
                // (4) retrieves representation
                // (5) retrieves unit
                // (6) retrieves display notation
                // (7) retrieves display precision
                // (8) demonstrates maximum and minimum imposition in a comment
                // (9) demonstrates value setter in a comment
                static void ExploreFloat(ArenaNET.IFloat fl)
                {
                        // Retrieve value
                        double value = fl.Value;

                        Console.WriteLine("{0}Value: {1}", TAB3, value);

                        // Retrieve range
                        double max = fl.Max;
                        double min = fl.Min;

                        Console.WriteLine("{0}Minimum, maximum,: {1}, {2}", TAB3, min, max);

                        // Retrieve increment
                        Boolean hasIncrement = fl.HasInc;

                        Console.Write("{0}Increment (mode): ", TAB3);

                        if(hasIncrement)
                        {
                                double inc = fl.Inc;
                                ArenaNET.EIncMode incMode = fl.IncMode;
                                String incModeStr = incMode.ToString();
                                Console.WriteLine("{0} {1}", inc, incModeStr);
                        }
                        else
                                Console.WriteLine("no increment");

                        // Retrieve representation
                        ArenaNET.ERepresentation rep = fl.Representation;
                        String repStr = rep.ToString();

                        Console.WriteLine("{0}Representation: {1}", TAB3, repStr);

                        // Retrieve unit
                        String unit = fl.Unit;

                        Console.WriteLine("{0}Unit: {1}", TAB3, unit);

                        // Retrieve display notation
                        ArenaNET.EDisplayNotation dispNotification = fl.DisplayNotation;
                        String dispNotificationStr = dispNotification.ToString();

                        Console.WriteLine("{0}Display notification: {1}", TAB3, dispNotificationStr);

                        // Retrieve display precision
                        long dispPrecision = fl.DisplayPrecision;

                        Console.WriteLine("{0}Display precision: {1}", TAB3, dispPrecision);

                        // Impose maximum and minimum
                        //    {
                        // long newMax = 10; long newMin = 0; fl.ImposeMax(newMax);
                        // fl.ImposeMin(newMin);
                        //    }

                        // Set value
                        //    {
                        // long value = 0; fl.Value = value;
                        //    }
                }

                // controls node exploration
                static void ExploreNodes(ArenaNET.INodeMap nodeMap)
                {
                        Console.Write("{0}\nInput node name to explore from Device Nodemap ('x' to exit)\n{1}", TAB1, TAB2);

                        // stay in loop until exit
                        while (true)
                        {
                                String nodeName = Console.ReadLine();

                                // exit manually on 'x'
                                if (nodeName == "x")
                                {
                                        Console.WriteLine("{0}Exit", TAB2);
                                        break;
                                }

                                try
                                {
                                        // get node
                                        ArenaNET.INode node = nodeMap.GetNode(nodeName);

                                        if (!node.Equals(null))
                                        {
                                                // explore by type
                                                switch (node.PrincipalInterfaceType)
                                                {
                                                        case ArenaNET.EInterfaceType.intfIBoolean:
                                                                ExploreBoolean(node as ArenaNET.IBoolean);
                                                                break;
                                                        case ArenaNET.EInterfaceType.intfIString:
                                                                ExploreString(node as ArenaNET.IString);
                                                                break;
                                                        case ArenaNET.EInterfaceType.intfIEnumeration:
                                                                ExploreEnumeration(node as ArenaNET.IEnumeration);
                                                                break;
                                                        case ArenaNET.EInterfaceType.intfIInteger:
                                                                ExploreInteger(node as ArenaNET.IInteger);
                                                                break;
                                                        case ArenaNET.EInterfaceType.intfIFloat:
                                                                ExploreFloat(node as ArenaNET.IFloat);
                                                                break;
                                                        default:
                                                                Console.WriteLine("{0}{1} type not found", TAB3, nodeName);
                                                                break;
                                                }
                                        }
                                }
                                catch(ArenaNET.RuntimeException)
                                {
                                        Console.WriteLine("{0}{1} not found", TAB3, nodeName);
                                }

                                // reset input
                                nodeName = "";
                                Console.Write("{0}\nInput node name to explore from Device Nodemap ('x' to exit)\n{1}", TAB1, TAB2);
                        }
                }

                // =-=-=-=-=-=-=-=-=-
                // =- PREPARATION -=-
                // =- & CLEAN UP =-=-
                // =-=-=-=-=-=-=-=-=-

                static void Main(string[] args)
                {
                        Console.WriteLine("Cs_Explore_NodeTypes");

                        try
                        {
                                // prepare example
                                ArenaNET.ISystem system = ArenaNET.Arena.OpenSystem();
                                system.UpdateDevices(100);
                                if (system.Devices.Count == 0)
                                {
                                        Console.WriteLine("\nNo camera connected\nPress enter to complete");
                                        Console.Read();
                                        return;
                                }
                                ArenaNET.IDevice device = system.CreateDevice(system.Devices[0]);
                                ArenaNET.INodeMap nodeMap = device.NodeMap;

                                // run example
                                Console.WriteLine("Commence example");
                                ExploreNodes(nodeMap);
                                Console.WriteLine("\nExample complete");

                                // clean up example
                                system.DestroyDevice(device);
                                ArenaNET.Arena.CloseSystem(system);
                        }
                        catch (Exception ex)
                        {
                                Console.WriteLine("Exception thrown: {0}", ex.Message);
                        }

                        Console.WriteLine("Press enter to complete");
                        Console.Read();
                }
        }
}
