using System;

namespace Cs_UserSets
{
        class Cs_UserSets
        {
                const String TAB1 = "  ";
                const String TAB2 = "    ";

                // User sets
                //    This example introduces user sets, a feature which allows for the
                //    saving and loading of multiple configurations. The example starts
                //    by changing two features, width and height. The device
                //    configuration is then saved to user set 1. The default user set is
                //    then loaded, followed by user set 1 again to demonstrate the the
                //    device configuration changing back and forth.

                // =-=-=-=-=-=-=-=-=-
                // =-=- SETTINGS =-=-
                // =-=-=-=-=-=-=-=-=-

                // width value to save/load
                const UInt32 WIDTH = 576;

                // height value to load/save
                const UInt32 HEIGHT = 512;

                // =-=-=-=-=-=-=-=-=-
                // =-=- EXAMPLE -=-=-
                // =-=-=-=-=-=-=-=-=-

                // demonstrates user sets
                // (1) gets width and height nodes
                // (2) changes width and height
                // (3) saves user set with width and height changes
                // (4) loads default user set, changing width and height to defaults
                // (5) loads user set, changing width and height back
                static void SaveAndLoadUserSets(ArenaNET.IDevice device)
                {
                        // Get width and height nodes
                        //    This example focuses on width and height nodes to demonstrate
                        //    the saving and loading of user sets.
                        Console.WriteLine("{0}Get width and height nodes", TAB1);

                        var widthNode = device.NodeMap.GetNode("Width") as ArenaNET.IInteger;
                        var heightNode = device.NodeMap.GetNode("Height") as ArenaNET.IInteger;

                        if (widthNode.Value.Equals(null) || !widthNode.IsReadable || !widthNode.IsWritable)
                                throw new ArenaNET.GenericException("Width node not found/readable/writable");

                        if (heightNode.Value.Equals(null) || !heightNode.IsReadable || !heightNode.IsWritable)
                                throw new ArenaNET.GenericException("Height node not found/readable/writable");

                        // change width and height
                        Console.WriteLine("{0}Change width and height", TAB1);

                        widthNode.Value = WIDTH;
                        heightNode.Value = HEIGHT;

                        // Save to user set 1
                        //    Saving the user set saves the new width and height values.
                        //    Saving a user set involves selecting the user set to save to on
                        //    the selector node and then executing the save on the command
                        //    node.
                        Console.WriteLine("{0}Save to user set 1", TAB1);

                        // select user set 1
                        var userSetSelectorNode = (ArenaNET.IEnumeration)device.NodeMap.GetNode("UserSetSelector");
                        userSetSelectorNode.FromString("UserSet1");

                        // execute the save
                        var userSetSaveNode = (ArenaNET.ICommand)device.NodeMap.GetNode("UserSetSave");
                        userSetSaveNode.Execute();

                        // check width and height
                        Console.WriteLine("{0}Width = {1}", TAB2, widthNode.Value);
                        Console.WriteLine("{0}Height = {1}", TAB2, heightNode.Value);

                        // Load default user set
                        //    Loading a user set is quite similar to saving: select the user
                        //    set to load using the selector node and then execute the load
                        //    using the command node. Notice that loading the default user
                        //    set changes the width and height values.
                        Console.WriteLine("{0}Load default user set", TAB1);

                        // select default user set
                        userSetSelectorNode.FromString("Default");

                        // execute the load
                        var userSetLoadNode = (ArenaNET.ICommand)device.NodeMap.GetNode("UserSetLoad");
                        userSetLoadNode.Execute();

                        // check width and height
                        Console.WriteLine("{0}Width = {1}", TAB2, widthNode.Value);
                        Console.WriteLine("{0}Height = {1}", TAB2, heightNode.Value);

                        // Load user set 1
                        //    Load user set 1 to return width and height to the values that
                        //    were saved earlier. The process is the same as above.
                        Console.WriteLine("{0}Load user set 1", TAB1);

                        // select user set 1
                        userSetSelectorNode.FromString("UserSet1");

                        // execute the load
                        userSetLoadNode.Execute();

                        // check width and height
                        Console.WriteLine("{0}Width = {1}", TAB2, widthNode.Value);
                        Console.WriteLine("{0}Height = {1}", TAB2, heightNode.Value);
                }

                // =-=-=-=-=-=-=-=-=-
                // =- PREPARATION -=-
                // =- & CLEAN UP =-=-
                // =-=-=-=-=-=-=-=-=-

                static void Main(string[] args)
                {
                        Console.WriteLine("Cs_UserSets");
                        Console.Write("Example may overwrite device settings saved to user set 1 -- proceed? ('y' to continue) ");
                        string continueExample = Console.ReadLine();

                        if (continueExample.Equals("y"))
                        {
                                try
                                {
                                        // prepare example
                                        ArenaNET.ISystem system = ArenaNET.Arena.OpenSystem();
                                        system.UpdateDevices(100);
                                        if (system.Devices.Count == 0)
                                        {
                                                Console.WriteLine("No camera connected\nPress enter to complete");
                                                Console.Read();
                                                return;
                                        }
                                        ArenaNET.IDevice device = system.CreateDevice(system.Devices[0]);

                                        // run example
                                        Console.WriteLine("Commence example\n");
                                        SaveAndLoadUserSets(device);
                                        Console.WriteLine("\nExample complete");

                                        // clean up example
                                        system.DestroyDevice(device);
                                        ArenaNET.Arena.CloseSystem(system);
                                }
                                catch (Exception ex)
                                {
                                        Console.WriteLine("Exception thrown: {0}", ex.Message);
                                }
                        }

                        Console.WriteLine("Press enter to complete");
                        Console.Read();
                }
        }
}
