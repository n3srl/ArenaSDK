Acquisition, Multithreaded Acquisition and Save¶

using System;
using System.Threading;

namespace Cs_Acquisition_MultithreadedAcquisitionAndSave
{

        class Cs_Acquisition_MultithreadedAcquisitionAndSave
        {
                const String TAB1 = "  ";
                const String TAB2 = "    ";

                // Acquiring and Saving Images on Seperate Threads: Introduction
                //    Saving images can sometimes create a bottleneck in the image
                //    acquisition pipeline. By sperating saving onto a separate thread,
                //    this bottle neck can be avoided. This example is programmed as a
                //    simple producer-consumer problem.

                // =-=-=-=-=-=-=-=-=-
                // =-=- SETTINGS =-=-
                // =-=-=-=-=-=-=-=-=-

                // image timeout
                const UInt32 TIMEOUT = 2000;

                // number of images to grab
                const UInt32 NUM_IMAGES = 25;

                // file name
                const string FILE_NAME = "Cs_Acquisition_MultithreadedAcquisitionAndSave/image";

                // file type
                const string FILE_TYPE = ".png";

                // pixel format
                const ArenaNET.EPfncFormat PIXEL_FORMAT = ArenaNET.EPfncFormat.BGR8;

                // =-=-=-=-=-=-=-=-=-
                // =-=- EXAMPLE -=-=-
                // =-=-=-=-=-=-=-=-=-

                // Images that are acquired from the device are stored in the queue.
                //    At the same time, images are removed from the queue and saved to
                //    disk. The only time we wait is when accessing the queue when the
                //    other thread is doing so, or when waiting for an image when the
                //    queue is empty. In that case, we simply lock thread.
                private static ProducerConsumer queue;

                static void AcquireImages(ArenaNET.IDevice device)
                {

                        // get initial values
                        var acquisitionModeNode = device.NodeMap.GetNode("AcquisitionMode") as ArenaNET.IEnumeration;
                        String acquisitionModeInitial = acquisitionModeNode.Symbolic;

                        Console.WriteLine("{0}Set acquisition mode to 'Continuous'\n", TAB1);
                        acquisitionModeNode.Symbolic = "Continuous";


                        // ensures the most recent image is delivered, even if it means
                        // skipping frames.
                        Console.WriteLine("{0}Set buffer handling mode to 'NewestOnly'", TAB1);

                        var streamBufferHandlingModeNode = (ArenaNET.IEnumeration)device.TLStreamNodeMap.GetNode("StreamBufferHandlingMode");
                        streamBufferHandlingModeNode.Symbolic = "NewestOnly";

                        // enable stream auto negotiate packet size
                        Console.WriteLine("{0}Enable stream to auto negotiate packet size", TAB1);

                        var streamAutoNegotiatePacketSizeNode = (ArenaNET.IBoolean)device.TLStreamNodeMap.GetNode("StreamAutoNegotiatePacketSize");
                        streamAutoNegotiatePacketSizeNode.Value = true;

                        // enable stream packet resend
                        Console.WriteLine("{0}Enable stream packet resend", TAB1);

                        var streamPacketResendEnableNode = (ArenaNET.IBoolean)device.TLStreamNodeMap.GetNode("StreamPacketResendEnable");
                        streamPacketResendEnableNode.Value = true;

                        // start stream
                        Console.WriteLine("{0}Start stream", TAB1);

                        device.StartStream();

                        // get images
                        Console.WriteLine("{0}Getting {1} images", TAB1, NUM_IMAGES);

                        for (UInt32 i = 0; i < NUM_IMAGES; i++)
                        {
                                // get image
                                Console.WriteLine("{0}Get image {1}", TAB2, i);

                                ArenaNET.IImage image = device.GetImage(TIMEOUT);

                                // get image information
                                Console.Write(" ({0} bytes; {1}x{2}; {3}; timestamp (ns): {4})",
                                        image.SizeFilled,
                                        image.Width,
                                        image.Height,
                                        image.PixelFormat.ToString(),
                                        image.TimestampNs);

                                // add image to queue
                                queue.Produce(ArenaNET.ImageFactory.Copy(image));

                                // requeue image buffer
                                Console.WriteLine("{0}Requeue image buffer", TAB1);

                                device.RequeueBuffer(image);
                        }

                        // Signal to other thread that no more images are incoming.
                        queue.completeTask();

                        // stop stream
                        Console.WriteLine("{0}Stop stream", TAB1);

                        device.StopStream();
                        // return nodes to their initial values
                        acquisitionModeNode.Symbolic = acquisitionModeInitial;
                }

                static void SaveImage()
                {
                        // Save images from queue
                        //    Pull images from queue and save them
                        int imageIndex = 0;
                        while (!queue.isTaskComplete())
                        {
                                var image = queue.Consume() as ArenaNET.IImage;
                                // convert image
                                Console.WriteLine("{0}Converting image {1} to {2}", TAB1, imageIndex, PIXEL_FORMAT);
                                ArenaNET.IImage imageConverted = ArenaNET.ImageFactory.Convert(image, PIXEL_FORMAT);

                                // prepare image parameters
                                SaveNET.ImageParams parameters = new SaveNET.ImageParams(
                                        imageConverted.Width,
                                        imageConverted.Height,
                                        imageConverted.BitsPerPixel,
                                        true);

                                // prepare image writer
                                Console.WriteLine("{0}Prepare image writer", TAB1);
                                string filename = FILE_NAME + imageIndex.ToString() + FILE_TYPE;
                                SaveNET.ImageWriter writer = new SaveNET.ImageWriter(parameters, filename);

                                // save image
                                writer.Save(imageConverted.DataArray, true);
                                ArenaNET.ImageFactory.Destroy(image);
                                ArenaNET.ImageFactory.Destroy(imageConverted);
                                imageIndex += 1;

                        }
                }


                // =-=-=-=-=-=-=-=-=-
                // =- PREPARATION -=-
                // =- & CLEAN UP =-=-
                // =-=-=-=-=-=-=-=-=-

                static void Main(string[] args)
                {
                        Console.WriteLine("Cs_Acquisition_MultithreadedAcquisitionAndSave");

                        try
                        {
                                // prepare example
                                ArenaNET.ISystem system = ArenaNET.Arena.OpenSystem();
                                system.UpdateDevices(100);
                                if (system.Devices.Count == 0)
                                {
                                        Console.WriteLine("\nNo camera connected\nPress enter to complete");
                                        Console.Read();
                                        return;
                                }

                                ArenaNET.IDevice device = system.CreateDevice(system.Devices[0]);

                                queue = new ProducerConsumer();
                                Thread t1 = new Thread(new ThreadStart(SaveImage));

                                // run example
                                Console.WriteLine("Commence example\n");
                                // Start thread that saves images
                                //    It will wait until images are added, then save them, until
                                //    we send the signal that no more images are incoming.
                                t1.Start();

                                // Adds images to queue
                                //    They will be saved by thread t1 as they are being added
                                AcquireImages(device);

                                // Wait for t1 to terminate
                                //    Termination signal will have been sent by AcquireImages
                                //    after all images have been sent to the queue and processed
                                t1.Join();
                                Console.WriteLine("\nExample complete");

                                // clean up example
                                system.DestroyDevice(device);
                                ArenaNET.Arena.CloseSystem(system);
                        }

                        catch (Exception ex)
                        {
                                Console.WriteLine("Exception thrown: {0}", ex.Message);
                        }

                        Console.WriteLine("Press enter to complete");
                        Console.Read();
                }
        }
        public class ProducerConsumer
        {
                // Queue shared between threads
                //    Has Produce and Consume functions to add and remove images Has
                //    isFinished to determine whether or not there are still images
                //    incoming The queue itself is shared, and can only be accessed by
                //    one thread at a time
                readonly object listLock = new object();
                readonly object boolLock = new object();
                bool isFinished = false;
                System.Collections.Queue queue = new System.Collections.Queue();

                public void Produce(object o)
                {
                        lock (listLock)
                        {
                                queue.Enqueue(o);
                                // We always need to pulse, even if the queue wasn't
                                //    empty before. Otherwise, if we add several items in quick
                                //    succession, we may only pulse once.
                                Monitor.Pulse(listLock);
                        }
                }

                public object Consume()
                {
                        lock (listLock)
                        {
                                // If the queue is empty, wait for an item to be added
                                //    this is a while loop, as we may be pulsed: we may be pulsed
                                //    but have a different thread come in and consume the object.
                                //    In that case, we'll have to wait for another pulse.
                                while (queue.Count == 0)
                                {
                                        // This releases listLock, only reacquiring it
                                        //    after being woken up by a call to Pulse
                                        Monitor.Wait(listLock);
                                }
                                return queue.Dequeue();
                        }
                }
                public bool isTaskComplete()
                {
                        // Note: may leave some Consume calls hanging
                        //    if there are multiple threads consuming from the queue
                        lock (boolLock)
                        {
                                return isFinished && queue.Count == 0;
                        }
                }

                public void completeTask()
                {
                        lock (boolLock)
                        {
                                isFinished = true;
                        }
                }
        }


}
