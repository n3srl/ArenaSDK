using System;
using System.Threading;

namespace Cs_Callback_Polling
{
        class Cs_Callback_Polling
        {
                const String TAB1 = "  ";
                const String TAB2 = "    ";
                const String TAB3 = "      ";

                // Callbacks: Polling
                //    This example demonstrates configuring a callback with polling.
                //    Polling allows for callbacks to be invoked over time.

                // =-=-=-=-=-=-=-=-=-
                // =-=- SETTINGS =-=-
                // =-=-=-=-=-=-=-=-=-

                // Number of times to poll
                const Int32 NUM_POLLS = 50;

                // Time to pass between polling (in milliseconds)
                const Int32 ELAPSED_TIME = 500;

                // callback counter
                static Int32 g_count = 0;

                // =-=-=-=-=-=-=-=-=-
                // =-=- EXAMPLE -=-=-
                // =-=-=-=-=-=-=-=-=-

                // callback function
                // (1) prints information from the callback
                static void PrintIntegerNodeValue(ArenaNET.INode node)
                {
                        // Do something on callback
                        //    This function is registered with the callback and invoked when
                        //    the node map has been polled if the elapsed time is over the
                        //    threshold of the node.
                        ArenaNET.IFloat deviceTemperatureNode = (ArenaNET.IFloat)node;

                        Console.Write("{0}  {1} Current device temperature: {2} 167 {3}\r", TAB3, g_count++.ToString().PadRight(2), deviceTemperatureNode.Value, deviceTemperatureNode.Unit);
                }

                // demonstrates polling
                // (1) gets the node map and node to poll
                // (2) registers callback
                // (3) starts stream
                // (4) polls node map at regular intervals, invoking callback
                // (5) stops stream
                // (6) deregisters callback
                static void ConfigureCallbackToPollDeviceTemperature(ArenaNET.IDevice device)
                {
                        // Get node map and device temperature node
                        //    Nodes are polled through their node maps. This example
                        //    demonstrates polling the device temperature node. It has a
                        //    polling time of 1 second, which means that its callback will
                        //    not be invoked within 1 second of the last time it has been
                        //    polled.
                        Console.WriteLine("{0}Get node map and device temperature node", TAB1);

                        ArenaNET.INodeMap nodeMap = device.NodeMap;
                        ArenaNET.INode node = nodeMap.GetNode("DeviceTemperature");

                        if (node == null || !node.IsReadable)
                                throw new ArenaNET.GenericException("Node not found/readable");

                        // Register callback
                        //    Callbacks are registered with a node and a function. This
                        //    example demonstrates callbacks being invoked through polling.
                        //    This happens when a node map is polled and the polling time is
                        //    ready to poll again.
                        Console.WriteLine("{0}Register callback", TAB1);

                        // create delegate for registering callback
                        ArenaNET.INode.del d = new ArenaNET.INode.del(PrintIntegerNodeValue);

                        IntPtr callback = node.RegisterCallback(d);

                        // enable stream auto negotiate packet size
                        var streamAutoNegotiatePacketSizeNode = (ArenaNET.IBoolean)device.TLStreamNodeMap.GetNode("StreamAutoNegotiatePacketSize");
                        streamAutoNegotiatePacketSizeNode.Value = true;

                        // enable stream packet resend
                        var streamPacketResendEnableNode = (ArenaNET.IBoolean)device.TLStreamNodeMap.GetNode("StreamPacketResendEnable");
                        streamPacketResendEnableNode.Value = true;

                        // start stream
                        Console.WriteLine("{0}Start stream", TAB1);

                        device.StartStream();

                        // Poll at regular intervals
                        //    The polling time of device temperature is 1 second. The
                        //    callback will only be invoked if the elapsed time since the
                        //    last callback is larger than the polling time. In this example,
                        //    two counts are shown: a polling count and a callback count.
                        //    This is to demonstrate that the callback is not necessarily
                        //    invoked every time the node map is polled.
                        Console.WriteLine("{0}Start polling", TAB1);

                        for (int i = 0; i < NUM_POLLS; i++)
                        {
                                Console.Write("{0} {1}\r", TAB2, i.ToString().PadRight(2));

                                // sleep for elapsed time
                                Thread.Sleep(ELAPSED_TIME);

                                // poll node on elapsed time
                                nodeMap.Poll(ELAPSED_TIME);
                        }
                        Console.WriteLine("{0}{1}", TAB2, NUM_POLLS);

                        // stop stream
                        Console.WriteLine("{0}Stop Stream", TAB1);

                        device.StopStream();

                        // Deregister callback
                        //    Failing to deregister a callback results in a memory leak. Once
                        //    a callback has been registered, it will no longer be invoked
                        //    when a node is invalidated.
                        Console.WriteLine("{0}Deregister callback", TAB1);

                        node.DeregisterCallback(callback);
                }

                // =-=-=-=-=-=-=-=-=-
                // =- PREPARATION -=-
                // =- & CLEAN UP =-=-
                // =-=-=-=-=-=-=-=-=-

                static void Main(string[] args)
                {
                        Console.WriteLine("Cs_Callback_Polling");

                        try
                        {
                                // prepare example
                                ArenaNET.ISystem system = ArenaNET.Arena.OpenSystem();
                                system.UpdateDevices(100);
                                if (system.Devices.Count == 0)
                                {
                                        Console.WriteLine("\nNo camera connected\nPress enter to complete");
                                        Console.Read();
                                        return;
                                }
                                ArenaNET.IDevice device = system.CreateDevice(system.Devices[0]);

                                // run example
                                Console.WriteLine("Commence example\n");
                                ConfigureCallbackToPollDeviceTemperature(device);
                                Console.WriteLine("\nExample complete");

                                // clean up example
                                system.DestroyDevice(device);
                                ArenaNET.Arena.CloseSystem(system);
                        }
                        catch (Exception ex)
                        {
                                Console.WriteLine("Exception thrown: {0}", ex.Message);
                        }

                        Console.WriteLine("Press enter to complete");
                        Console.Read();
                }
        }
}
